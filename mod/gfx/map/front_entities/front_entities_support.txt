### Aerial Recon ###

REPLACE_OR_CREATE:generic_aerial_bleriotxi_01_diorama_entity = {
	entity = "generic_aerial_bleriotxi_01_entity"

	group = "cavalry"

	trigger = {
		OR = {
		scope:combat_unit_type = unit_type:combat_unit_type_fixed_wing_aircraft
		scope:combat_unit_type = unit_type:combat_unit_type_multirole_aircraft
		scope:combat_unit_type = unit_type:combat_unit_type_stealth_aircraft
		scope:combat_unit_type = unit_type:combat_unit_type_uav
		scope:combat_unit_type = unit_type:combat_unit_type_autonomous_aircraft
		}
	}
}

		scope:culture = {
