nuclear_disarmament = {
	icon = "gfx/interface/icons/war_goals/contain_threat.dds"

	kind = contain_threat

	settings = {
		require_target_be_part_of_war
	}

	execution_priority = 6

	contestion_type = control_target_country_capital

	target_type = country

	possible = {
		# Initiator must be a great or major power
		ROOT = {
			country_rank >= rank_value:major_power
		}
		scope:target_country ?= {
			OR = {
				has_variable = nuclear_enrichment_active
				has_variable = nuclear_capable
				has_variable = nuclear_production_active
				has_variable = nukes
			}
		}
	}

	valid = {
		ROOT = {
			country_rank >= rank_value:major_power
		}
		scope:target_country ?= {
			OR = {
				has_variable = nuclear_enrichment_active
				has_variable = nuclear_capable
				has_variable = nuclear_production_active
				has_variable = nukes
			}
		}
	}

	maneuvers = {
		value = 0

		add = {
			desc = "MANEUVERS_BASE_VALUE"
			value = 15
		}

		scope:target_country = {
			multiply = {
				desc = "MANEUVERS_TARGET_COUNTRY_POPULATION_FACTOR"
				value = total_population
				divide = define:NDiplomacy|SWAY_OFFER_WARGOAL_MANEUVERS_COST_POPULATION_SCALING_FACTOR
				multiply = define:NDiplomacy|SWAY_OFFER_WARGOAL_MANEUVERS_COST_POPULATION_SCALING_MULTIPLIER
				max = define:NDiplomacy|SWAY_OFFER_WARGOAL_MANEUVERS_COST_POPULATION_SCALING_MULTIPLIER_MAX_TOTAL
			}
		}
	}

	infamy = {
		value = 0

		add = {
			desc = "INFAMY_BASE_VALUE"
			value = 0.5
		}

		scope:target_country = {
			multiply = {
				desc = "INFAMY_TARGET_COUNTRY_POPULATION_FACTOR"
				add = {
					every_scope_state = {
						add = {
							value = state_population
							divide = define:NDiplomacy|WAR_GOAL_INFAMY_POPULATION_SCALING_FACTOR
							multiply = define:NDiplomacy|WAR_GOAL_INFAMY_POPULATION_SCALING_MULTIPLIER
							multiply = {
								if = {
									limit = { is_incorporated = no }
									value = 1
									add = define:NDiplomacy|WAR_GOAL_INFAMY_UNINCORPORATED_FACTOR
								}
							}
							max = define:NDiplomacy|WAR_GOAL_INFAMY_POPULATION_SCALING_MULTIPLIER_MAX_PER_STATE
						}
					}
				}
				add = 1
				max = define:NDiplomacy|WAR_GOAL_INFAMY_POPULATION_SCALING_MULTIPLIER_MAX_TOTAL
			}
		}

		min = {
			desc = "INFAMY_MINIMUM_VALUE"
			value = 1
		}
	}

	on_enforced = {
		scope:target_country ?= {
			# ── STRIP THE ARSENAL ────────────────────
			# Warhead count
			remove_variable = nukes

			# ── STRIP PROGRAMME STATUS ───────────────
			# Enrichment
			remove_variable = nuclear_enrichment_active
			remove_variable = uranium_enrichment_progress
			remove_variable = nuclear_enrichment_active

			# Capability
			remove_variable = nuclear_capable

			# Production
			remove_variable = nuclear_production_active
			remove_variable = nuclear_production_progress

			# Exposure / diplomatic flags
			remove_variable = nuclear_programme_exposed
			remove_variable = nuclear_programme_denied
			remove_variable = nuclear_programme_acknowledged
			remove_variable = nuclear_skipped_iaea
			remove_variable = nuclear_iaea_refused
			remove_variable = nuclear_un_sanctions_active
			remove_variable = nuclear_complied_with_un

			# Deployment settings
			remove_variable = nuclear_deployment_percentage
			remove_variable = nuclear_kill_rate_cache

			# ── CLEAR TARGET LISTS ───────────────────
			clear_variable_list = wmd_target_list
			clear_variable_list = wmd_target_list_c
			clear_variable_list = nuclear_enemy_list
			clear_variable_list = nuclear_enemy_states_list

			# ── REMOVE NUCLEAR MODIFIERS ─────────────
			remove_modifier = nuclear_capable_modifier
			remove_modifier = nuclear_programme_denied
			remove_modifier = nuclear_programme_acknowledged
			remove_modifier = nuclear_warhead_produced_modifier
			remove_modifier = nuclear_un_sanctions
			remove_modifier = nuclear_iaea_defiance
			remove_modifier = nuclear_military_ultimatum
			remove_modifier = nuclear_great_power_pressure

			# ── POLITICAL CONSEQUENCES ───────────────
			# The disarmed country pays a heavy domestic price
			country_add_legitimacy = -20
			add_modifier = {
				name = nuclear_disarmament_enforced
				months = 120
			}
			every_scope_state = {
				limit = { is_incorporated = yes }
				add_radicals_in_state = { value = large_radicals }
			}

			# Notify the target country
			trigger_event = { id = nuclear_events.10 popup = yes }
		}

		# The enforcing country gets prestige for non-proliferation
		ROOT = {
			add_modifier = {
				name = nuclear_disarmament_enforcer
				months = 60
			}
			country_add_legitimacy = 10
		}
	}

	ai = {
		is_significant_demand = yes
	}
}
