 
nuclear_enrichment_progress_bar = {
	name = "nuclear_enrichment_progress_bar"
	desc = "nuclear_enrichment_progress_bar_desc"
 
	start_value = 0
	min_value = 0
	max_value = 100
 
	default = yes
 
	monthly_progress = {
		if = {
			limit = { exists = var:uranium_enrichment_progress }
			add = {
				desc = "nuclear_enrichment_progress_bar_tt"
				value = var:uranium_enrichment_progress
			}
		}
	}
}
