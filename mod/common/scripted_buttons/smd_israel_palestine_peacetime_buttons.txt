
je_ip_peacetime_security_coordination_button = {
	name = "je_ip_peacetime_security_coordination"
	desc = "je_ip_peacetime_security_coordination_desc"

	visible = {
		OR = {
			c:ISR ?= this
			c:PAL ?= this
		}
	}

	possible = {
		hidden_trigger = {
			NOT = { has_variable = ip_security_coordination_cooldown }
		}
	}

	selected = {
		has_modifier = ip_security_coordination
	}

	effect = {
		set_variable = { name = ip_security_coordination_cooldown months = 12 }
		add_modifier = { name = ip_security_coordination months = 18 }

		change_global_variable = { name = ip_peacetime_security_coordination add = 12 }
		ip_temperature_subtract_8 = yes

		# Security coordination also creates domestic backlash.
		if = {
			limit = { c:PAL ?= this }
			add_modifier = { name = ip_political_backlash months = 12 }
		}
		else_if = {
			limit = { c:ISR ?= this }
			change_global_variable = { name = ip_peacetime_international_pressure add = -2 }
		}
	}

	ai_chance = {
		value = 20
		if = {
			limit = {
			je:je_israel_palestine_peacetime ?= {
				scripted_bar_progress = {
					name = je_israel_palestine_tension_bar
					value >= 0.70
				}
			}
		}
			add = 50
		}
	}
}
