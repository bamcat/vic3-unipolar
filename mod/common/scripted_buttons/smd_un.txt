sb_request_peacekeepers = {
	name = "sb_request_peacekeepers"
	desc = "sb_request_peacekeepers_desc"

	visible = {
		always = yes
	}

	cooldown = { days = 730 }
	
	possible = {
		is_at_war = yes
		country_turmoil > 0.08
		custom_tooltip = {
		NOT = { global_var:un_peacekeeping_num_votes >= 1 }
		NOT = { global_var:un_peacekeeping_num_votes <= -1 }
		global_var:un_peacekeeping_num_votes = 0
		text = un_already_voting_on_peacekeeping
		}
		OR = { 
		NOT = { has_modifier = un_peacekeepers_deployed }
		NOT = { has_modifier = un_peacekeepers_vetoed }
		}
		random_diplomatic_play = {
			limit = { is_war = yes any_scope_play_involved = this }
			has_war_support = {
			target = c:this
			value < 50
			}
		}
	}

	ai_chance = {
		value = 25
		if = { limit = { c:SRB ?= this }
			add = -25
		}
	}
	
	effect = {
		trigger_event = { id = un_events.1 popup = yes }
	}
}

# ── VOTE YAY ─────────────────────────────
# Visible only when a motion is active
# and this country has not yet voted.
 
un_vote_yay_button = {
	name = "un_vote_yay_button"
	desc = "un_vote_yay_button_desc"
 
	visible = {
		has_global_variable = un_motion_active
		NOT = { has_variable = un_voted_on_motion }
	}
 
	possible = {
		always = yes
	}
 
	ai_chance = {
		value = 60
		modifier = {
			trigger = {
				OR = {
					c:USA ?= this
					c:GBR ?= this
					c:FRA ?= this
				}
			}
			add = 20
		}
	}
 
	effect = {
		set_variable = un_voted_on_motion
		if = {
			limit = {
				OR = {
					c:USA ?= this
					c:FRA ?= this
					c:GBR ?= this
					c:RUS ?= this
					c:PRC ?= this
				}
			}
			# Permanent members get double weight
			je:je_united_nations = {
				add_progress = {
				value = 2
				name = un_motion_progress_bar
					}
				}
		}
		else = {
			je:je_united_nations = {
				add_progress = {
				value = 1
				name = un_motion_progress_bar
					}
				}
			}
	}
}
