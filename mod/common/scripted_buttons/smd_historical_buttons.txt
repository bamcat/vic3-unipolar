opec_propose_increase_button = {
	name = "opec_propose_increase_button"
	desc = "opec_propose_increase_button_desc"
	cooldown = { days = 365 }

	visible = {
		NOT = { has_global_variable = opec_vote_active }
	}

	possible = {
		is_target_in_global_variable_list = { name = io_opec_members target = this }
		custom_tooltip = {
			text = opec_no_vote_active_tt
			NOT = { has_global_variable = opec_vote_active }
		}
		NOT = { has_modifier = opec_increased_production }
	}

	ai_chance = {
		value = 10
		# AI more likely to propose increase when oil is expensive
		modifier = {
			trigger = {
				market = { mg:oil = { market_goods_pricier > 0.25 } }
			}
			add = 40
		}
		# Less likely when oil is cheap
		modifier = {
			trigger = {
				market = { mg:oil = { market_goods_cheaper > 0.10 } }
			}
			add = -15
		}
	}

	effect = {
		set_global_variable = { name = opec_vote_active value = yes }
		set_global_variable = { name = opec_vote_type value = flag:opec_increase }
		set_global_variable = { name = opec_yes_votes value = 0 }
		set_global_variable = { name = opec_no_votes value = 0 }
		# Clear prior vote flags
		every_country = {
			limit = { has_variable = opec_voted }
			remove_variable = opec_voted
		}
		# Notify all OPEC members
		every_country = {
			limit = { is_target_in_global_variable_list = { name = io_opec_members target = this } }
			trigger_event = { id = opec_events.1 days = 0 popup = yes }
		}
		# Schedule result after 30 days
		trigger_event = { id = opec_events.10 days = 30 }
	}
}

### VOTE YAY — visible when vote is active and not yet voted
### ══════════════════════════════════════════

opec_vote_yay_button = {
	name = "opec_vote_yay_button"
	desc = "opec_vote_yay_button_desc"

	visible = {
		has_global_variable = opec_vote_active
		NOT = { has_variable = opec_voted }
		is_target_in_global_variable_list = { name = io_opec_members target = this }
	}

	possible = {
		always = yes
	}

	ai_chance = {
		value = 50
		# For increase votes: AI favors if oil is expensive (they can sell more)
		modifier = {
			trigger = {
				global_var:opec_vote_type = flag:opec_increase
				market = { mg:oil = { market_goods_pricier > 0.15 } }
			}
			add = 30
		}
		# For decrease votes: AI favors if oil is cheap (restrict supply)
		modifier = {
			trigger = {
				global_var:opec_vote_type = flag:opec_decrease
				market = { mg:oil = { market_goods_cheaper > 0.10 } }
			}
			add = 30
		}
		# For embargo votes: AI favors if target has high infamy
		modifier = {
			trigger = {
				global_var:opec_vote_type = flag:opec_embargo
				has_global_variable = opec_embargo_target
				global_var:opec_embargo_target = {
					infamy >= infamy_threshold:pariah
				}
			}
			add = 30
		}
		# AI less likely to embargo if target is friendly
		modifier = {
			trigger = {
				global_var:opec_vote_type = flag:opec_embargo
				has_global_variable = opec_embargo_target
				has_attitude = { who = global_var:opec_embargo_target attitude = attitude_friendly }
			}
			add = -50
		}
	}

	effect = {
		set_variable = opec_voted
		je:je_opec = {
			add_progress = {
				value = 1
				name = opec_vote_progress_bar
			}
		}
	}
}
