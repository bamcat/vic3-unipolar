# Second American Civil War JE diplomacy buttons

2acw_offer_new_africa_blue_button = {
	scope = country

	is_shown = {
		c:BLA ?= ROOT
		has_global_variable = second_american_civil_war_started
		exists = c:NAF
	}

	is_valid = {
		c:NAF ?= {
			NOT = { has_variable = 2acw_blue_aligned }
			NOT = { has_variable = 2acw_declined_blue_offer }
		}
	}

	effect = {
		custom_tooltip = { text = 2acw_offer_new_africa_blue_tt }
		c:NAF ?= {
			trigger_event = { id = great_polarization.120 popup = yes }
		}
	}
}
