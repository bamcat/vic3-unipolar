je_kurdistan_war_spring_offensive_button = {
	name = "je_kurdistan_war_spring_offensive"
	desc = "je_kurdistan_war_spring_offensive_desc"

	visible = {
		c:KUR ?= this
	}

	possible = {
		c:KUR ?= THIS
		hidden_trigger = {
			NOT = { has_global_variable = kurdistan_war_final_phase }
		}
		OR = {
			custom_tooltip = {
				text = "KURDISTAN_WAR_OFFENSIVE_MINIMUM_SCORE"
				global_var:kur_kurdistan_war_score >= kurdistan_war_offensive_min_score
			}
			game_date >= 1993.1.1
		}
	}

	selected = {
		has_global_variable = kurdistan_war_final_phase
	}

	effect = {
		je:je_kurdistan_war ?= {
			remove_modifier = kurdistan_war_guerilla_warfare
		}
		set_global_variable = kurdistan_war_final_phase
		remove_global_variable = kurdistan_war_skirmish_phase
		trigger_event = { id = kurdistan_war.1 popup = yes }
		hidden_effect = { c:TRK ?= { trigger_event = { id = kurdistan_war.1 popup = yes } } }

		add_modifier = {
			name = kur_mountain_resistance
			multiplier = {
				value = global_var:kur_kurdistan_war_score
				min = 1
			}
			years = 1
			is_decaying = yes
		}
		c:TRK ?= {
			add_modifier = {
				name = trk_security_operation
				multiplier = {
					value = global_var:trk_kurdistan_war_score
					min = 1
				}
				years = 2
				is_decaying = yes
			}
		}
	}

	ai_chance = {
		value = 0

		if = {
			limit = { global_var:kur_kurdistan_war_score >= 10 }
			add = 10
		}
		if = {
			limit = { global_var:kur_kurdistan_war_score >= 15 }
			add = 10
		}
		if = {
			limit = { global_var:kur_kurdistan_war_score >= 20 }
			add = 10
		}
		if = {
			limit = { game_date > 1993.1.1 }
			add = 10000
		}
	}
}
