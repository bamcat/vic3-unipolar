CUB = { # Cuba
	# Revolution: communist/vanguardist
	dynamic_country_name = {
		name = dyn_c_cuba_peoples_republic
		adjective = CUB_ADJ

		is_revolutionary = yes
		priority = 1

		trigger = {
			exists = scope:actor
			scope:actor = {
				is_secessionist = no
				any_interest_group = {
					is_in_government = yes
					is_powerful = yes
					leader = {
						OR = {
							has_ideology = ideology_communist
							has_ideology = ideology_vanguardist
							has_ideology = ideology_anarchist
						}
					}
				}
			}
		}
	}
	# Revolution: fascist / ethno-nationalist
	dynamic_country_name = {
		name = dyn_c_cuba_national_state
		adjective = CUB_ADJ

		is_revolutionary = yes
		priority = 1

		trigger = {
			exists = scope:actor
			scope:actor = {
				is_secessionist = no
				any_interest_group = {
					is_in_government = yes
					is_powerful = yes
					leader = {
						OR = {
							has_ideology = ideology_fascist
							has_ideology = ideology_ethno_nationalist
						}
					}
				}
			}
		}
	}
	# Revolution: any other (generic liberal/democratic/military)
	dynamic_country_name = {
		name = dyn_c_cuba_republic
		adjective = CUB_ADJ

		is_revolutionary = yes
		priority = 0

		trigger = {
			exists = scope:actor
			scope:actor = {
				is_secessionist = no
			}
		}
	}
	# Non-revolution: republic government
	dynamic_country_name = {
		name = dyn_c_cuba_republic
		adjective = CUB_ADJ

		is_main_tag_only = yes
		priority = 5

		trigger = {
			coa_def_republic_flag_trigger = yes
			NOR = {
				has_law = law_type:law_council_republic
				has_law = law_type:law_one_party_state
				has_law = law_type:law_single_party_state
			}
		}
	}
}
