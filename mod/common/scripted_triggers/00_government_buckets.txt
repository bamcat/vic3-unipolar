	OR = {
		smd_is_theocratic_government_trigger = yes
		smd_is_left_authoritarian_government_trigger = yes
		smd_is_right_authoritarian_government_trigger = yes
		smd_is_left_democratic_government_trigger = yes
		smd_is_right_democratic_government_trigger = yes
	}
}

# Safety/debug helper: if this returns yes, the country is using a government type not listed above.
# Moderates remain always-valid in the ideology file, so even an unknown government type will not block all ideology spawning.
smd_has_unknown_government_classification_trigger = {
	NOT = { smd_has_any_government_classification_trigger = yes }
}
