# ── UNITED STATES ─────────────────────────

# MEDICAID
# Federal health insurance for low-income
# Americans. Parent: charitable or no health
# system — fills the gap left by private
# markets.

amendment_medicaid = {
	parent = law_private_health_insurance
	allowed_laws = {
		law_no_health_system
		law_charitable_health_system
		law_private_health_insurance
		law_public_health_insurance
	}
	modifier = {
		# Medicaid expands healthcare access for the poor
		state_mortality_mult                        = -0.01
		state_mortality_wealth_mult                 = -0.01
		state_lower_strata_standard_of_living_add   = 0.25
	}
	possible = {
		c:USA ?= this
	}
	ai_will_revoke = {
		NOT = {
			any_interest_group = {
				is_in_government = yes
				OR = {
					is_interest_group_type = ig_trade_unions
					is_interest_group_type = ig_intelligentsia
					is_interest_group_type = ig_petty_bourgeoisie
					is_interest_group_type = ig_professionals
				}
			}
		}
	}
	can_repeal = {
		legitimacy >= legitimacy_to_repeal_amendment
	}
	amendment_activism_multiplier = 0.75
}
