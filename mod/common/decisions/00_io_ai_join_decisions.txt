io_ai_request_join_nato = {
	is_shown = {
		is_ai = yes
		NOT = { is_target_in_global_variable_list = { name = io_nato_members target = this } }
		OR = {
			capital = { is_in_geographic_region = geographic_region_europe }
			capital = { is_in_geographic_region = geographic_region_north_america }
			c:TRK ?= this
		}
	}

	possible = {
		is_ai = yes
		NOT = { is_target_in_global_variable_list = { name = io_nato_members target = this } }
		NOT = { has_variable = io_pending_org }
		NOT = { has_variable = io_cooldown_nato }
		NOT = { has_variable = io_ai_application_cooldown }
		NOT = { has_global_variable = io_ai_nato_application_global_cooldown }
		NOT = { io_has_authoritarian_government_laws = yes }
		OR = {
			capital = { is_in_geographic_region = geographic_region_europe }
			capital = { is_in_geographic_region = geographic_region_north_america }
			c:TRK ?= this
		}
	}

	when_taken = {
		io_ai_apply_to_nato = yes
	}

	ai_chance = {
		base = 1
		modifier = {
			trigger = {
				OR = {
					has_attitude = { who = c:USA attitude = attitude_friendly }
					has_attitude = { who = c:USA attitude = attitude_amicable }
					has_attitude = { who = c:USA attitude = attitude_cooperative }
				}
			}
			add = 4
		}
		modifier = {
			trigger = { legitimacy > 60 }
			add = 2
		}
	}

	cooldown = { days = 365 }
}
