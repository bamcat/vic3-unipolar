dissolve_european_union = {

	is_shown = {
		NOT = { c:EUR ?= this }
		any_scope_state = {
			state_region = s:STATE_FLANDERS
		}
	}
 
	possible = {
		# No EEC members remain
		custom_tooltip = {
			text = io_dissolve_eu_no_members_tt
			NOT = {
				any_country = {
					OR = {
						is_target_in_global_variable_list = { name = io_eec_members target = this }
						is_target_in_global_variable_list = { name = io_eu_members target = this }
					}
				}
			}
		}
	}
 
	when_taken = {
		# Annex EUR if it still exists
		if = {
			limit = {
				c:EUR ?= { exists = yes }
			}
			c:EUR = {
				every_scope_state = {
					set_state_owner = ROOT
				}
			}
		}
		# Clear any remaining stale entries from both lists
		every_country = {
			limit = { is_target_in_global_variable_list = { name = io_eec_members target = this } }
			remove_list_global_variable = { name = io_eec_members target = this }
			remove_modifier = io_free_trade_agreement_member
		}
		every_country = {
			limit = { is_target_in_global_variable_list = { name = io_eu_members target = this } }
			remove_list_global_variable = { name = io_eu_members target = this }
			remove_modifier = io_free_trade_agreement_member
			remove_modifier = modifier_eu_member_state
			remove_variable = eu_member_state
		}
		set_global_variable = io_eu_dissolved
		set_global_variable = io_eec_dissolved
		custom_tooltip = {
			text = io_dissolve_eu_effect_tt
		}
	}
 
	ai_chance = {
		base = 0
	}
}
