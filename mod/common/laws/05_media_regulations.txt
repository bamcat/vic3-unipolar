# ── FAIRNESS DOCTRINE ─────────────────────
# Media outlets must present balanced
# coverage of political issues.
# Radical movements lose amplification.
# State authority stabilises.

law_fairness_doctrine = {
	group = lawgroup_media_regulations
	icon = "gfx/interface/icons/law_icons/smd/fairness_doctrine.dds"

	progressiveness = 100

	modifier = {
		# Balanced coverage dampens radical movement recruitment
		country_authority_mult                        = 0.05
		country_legitimacy_base_add                   = 3
		state_radicals_from_political_movements_mult  = -0.20
		country_prestige_mult                         = 0.05
		building_art_academy_throughput_add = 0.5
	}

	ai_will_do = {
		NOR = {
			has_law = law_type:law_government_content_moderators
			has_law = law_type:law_state_wifi
		}
		has_law_or_variant = law_type:law_digital_rights
	}

	ai_enact_weight_modifier = {
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		if = {
			limit = {
				has_law = law_type:law_digital_rights
				literacy_rate >= 0.40
			}
			add = 50
		}
	}
}

# ── GOVERNMENT CONTENT MODERATORS ─────────
# The state actively moderates online
# content — removing disinformation,
# extremist material, and content deemed
# harmful. A powerful check on radical
# movements at the cost of legitimacy
# concerns about censorship.

law_government_content_moderators = {
	group = lawgroup_media_regulations
	icon = "gfx/interface/icons/law_icons/smd/content_moderators.dds"

	progressiveness = 75

	modifier = {
		state_radicals_from_political_movements_mult  = -0.25
		state_loyalists_from_political_movements_mult = 0.05
		country_authority_mult                        = 0.10
		building_art_academy_throughput_add = 0.25
	}

	ai_will_do = {
		NOT = { has_law = law_type:law_state_wifi }
		has_law_or_variant = law_type:law_fairness_doctrine
		OR = {
			has_law = law_type:law_autocracy
			has_law = law_type:law_single_party_state
			has_law = law_type:law_oligarchy
		}
	}

	ai_enact_weight_modifier = {
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		if = {
			limit = {
				has_law = law_type:law_fairness_doctrine
				OR = {
					has_law = law_type:law_autocracy
					has_law = law_type:law_single_party_state
				}
			}
			add = 100
		}
	}
}
