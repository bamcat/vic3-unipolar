# JUDICIARY
# ── INQUISITIVE ───────────────────────────

law_inquisitive_judiciary = {
	group = lawgroup_judiciary
	icon = "gfx/interface/icons/law_icons/smd/inquisitive.dds"

	progressiveness = 0

	modifier = {
		country_authority_mult                      = 0.05
		country_legitimacy_base_add                 = 5
		country_law_enactment_success_add           = 0.05
		country_law_enactment_imposition_success_add = 0.05
		# Efficient prosecution suppresses turmoil
		state_turmoil_effects_mult                  = -0.05
		country_allow_enacting_decrees_in_subject_bool = yes
	}

	institution = institution_courts
	institution_modifier = {
		state_turmoil_effects_mult                  = -0.03
		country_revolution_progress_mult            = -0.025
	}

	ai_will_do = {
		always = yes
	}

	ai_enact_weight_modifier = {
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		if = {
			limit = {
				OR = {
					has_law = law_type:law_autocracy
					has_law = law_type:law_single_party_state
				}
			}
			add = 100
		}
	}
}

# ── CUSTOMARY ─────────────────────────────
# Disputes are resolved through traditional
# community norms, tribal councils, and
# local custom rather than formal courts.
# No institution — customary law predates
# and resists state structures.

law_customary_judiciary = {
	group = lawgroup_judiciary
	icon = "gfx/interface/icons/law_icons/smd/customary.dds"

	progressiveness = 20

	modifier = {
		# Communities regulate themselves — less state authority
		country_authority_mult                      = -0.05
		# But deep social legitimacy from respected traditions
		country_legitimacy_base_add                 = 10
		country_loyalists_from_legitimacy_mult      = 0.10
		# Lower strata benefit most from accessible local justice
		state_lower_strata_standard_of_living_add   = 0.25
		state_turmoil_effects_mult                  = -0.05
	}

	ai_will_do = {
		NOR = {
			has_law = law_type:law_religious_judiciary
			has_law = law_type:law_restorative_judiciary
			has_law = law_type:law_adversarial_judiciary
		}
	}

	ai_enact_weight_modifier = {
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		if = {
			limit = { literacy_rate < 0.25 }
			add = 75
		}
	}
}

# ── RELIGIOUS LAW ─────────────────────────
# Scripture and religious jurisprudence
# govern legal proceedings. The devout
# IG gains enormous political strength.
# Strong for theocratic states.

law_religious_judiciary = {
	group = lawgroup_judiciary
	icon = "gfx/interface/icons/law_icons/smd/religious_law.dds"

	progressiveness = 40

	modifier = {
		country_clergymen_pol_str_mult              = 0.20
		# Religious authority grants deep legitimacy
		country_legitimacy_base_add                 = 5
		country_loyalists_from_legitimacy_mult      = 0.10
		# Scripture as law is stable but inflexible
		country_authority_mult                      = 0.05
		country_law_enactment_success_add           = -0.05
		# Non-believers and minorities face discrimination
		country_political_strength_second_rate_citizen_mult = -0.20
		country_political_strength_open_prejudice_mult = -0.20
		country_political_strength_violent_hostility_mult = -0.20
		country_political_strength_cultural_erasure_mult = -0.20
		country_institution_cost_institution_courts_mult = -0.10
	}

	institution = institution_courts
	institution_modifier = {
		interest_group_ig_devout_pol_str_mult       = 0.10
	}

	ai_will_do = {
		NOR = {
			has_law = law_type:law_restorative_judiciary
			has_law = law_type:law_adversarial_judiciary
		}
		has_law = law_type:law_state_religion
	}

	ai_enact_weight_modifier = {
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		if = {
			limit = {
				has_law = law_type:law_state_religion
				any_interest_group = {
					is_interest_group_type = ig_devout
					is_in_government = yes
					is_powerful = yes
				}
			}
			add = 150
		}
	}
}

# ── RESTORATIVE ───────────────────────────

law_restorative_judiciary = {
	group = lawgroup_judiciary
	icon = "gfx/interface/icons/law_icons/smd/restorative_justice.dds"

	progressiveness = 60

	modifier = {
		# Rehabilitation reduces turmoil and mortality
		state_turmoil_effects_mult                  = -0.05
		state_mortality_turmoil_mult                = -0.04
		# Reconciliation processes build loyalty
		country_loyalists_from_legitimacy_mult      = 0.10
		country_radicals_from_legitimacy_mult       = -0.05
		# Less punitive
		country_authority_mult                      = -0.03
		interest_group_ig_intelligentsia_pol_str_mult = 0.45
	}

	institution = institution_courts
	institution_modifier = {
		state_loyalists_from_political_movements_mult = 0.05
		state_mortality_mult                        = -0.01
		state_education_access_add                  = 0.025
	}

	ai_will_do = {
		NOT = { has_law = law_type:law_adversarial_judiciary }
		has_law_or_variant = law_type:law_customary_judiciary
		OR = {
			has_law = law_type:law_universal_suffrage
			has_law = law_type:law_census_voting
		}
	}

	ai_enact_weight_modifier = {
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		if = {
			limit = {
				has_law = law_type:law_universal_suffrage
				literacy_rate >= 0.50
			}
			add = 75
		}
	}
}
