law_total_prohibition = {
	group = lawgroup_drug_regulations
	icon = "gfx/interface/icons/law_icons/smd/total_prohibition.dds"

	progressiveness = 0

	modifier = {
		country_authority_mult                              = 0.05
		country_legitimacy_base_add                        = 5
		country_turmoil_effects_mult                       = 0.10
		state_mortality_mult                               = 0.02
		building_opium_plantation_throughput_add           = -0.25
		interest_group_ig_devout_pol_str_mult              = 0.10
	}

	ai_will_do = {
		OR = {
			has_law = law_type:law_total_prohibition
			has_law = law_type:law_alcohol_tobacco_legal
		}
	}

	ai_enact_weight_modifier = {
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		# Authoritarian governments more likely to prohibit
		if = {
			limit = {
				OR = {
					has_law = law_type:law_autocracy
					has_law = law_type:law_single_party_state
				}
			}
			add = 100
		}
		# Devout IG in government pushes for prohibition
		if = {
			limit = {
				any_interest_group = {
					is_interest_group_type = ig_devout
					is_in_government = yes
					is_powerful = yes
				}
			}
			add = 150
		}
	}
}
