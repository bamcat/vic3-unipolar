# ── FRACTIONAL RESERVE BANKING ────────────
# Banks hold only a fraction of deposits as
# reserves, lending the rest. Securities
# production is multiplied as financial
# districts operate with high leverage.
# Credit production is debuffed — banks
# are expanding the money supply rather
# than simply storing it.
# No institution — this is the default
# modern banking arrangement.

law_fractional_reserve_banking = {
	group = lawgroup_central_banking
	icon = "gfx/interface/icons/law_icons/smd/fractional_reserve_banking.dds"

	progressiveness = 50
	
	modifier = {
	}

	ai_will_do = {
		NOT = { has_law = law_type:law_full_reserve_banking }
	}

	ai_enact_weight_modifier = {
		value = 100

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}

# ── FULL RESERVE BANKING ──────────────────
# Banks must hold 100% of deposits as
# reserves. No fractional lending.
# Unlocks institution_central_bank.
# Each institution level raises the interest
# rate by 1% but increases the maximum
# credit limit by 5%.

law_full_reserve_banking = {
	group = lawgroup_central_banking
	icon = "gfx/interface/icons/law_icons/smd/full_reserve_banking.dds"

	progressiveness = 100

	modifier = {
		building_group_bg_banking_throughput_add    = 0.10
		building_financial_district_throughput_add  = -0.10
		country_loan_interest_rate_add              = -0.01
	}

	ai_will_do = {
		has_law_or_variant = law_type:law_fractional_reserve_banking
		OR = {
			has_law = law_type:law_interventionism
			has_law = law_type:law_command_economy
		}
	}

	ai_enact_weight_modifier = {
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		if = {
			limit = {
				has_law = law_type:law_fractional_reserve_banking
				has_law = law_type:law_interventionism
				scaled_debt >= 0.50
			}
			add = 100
		}
	}
}
