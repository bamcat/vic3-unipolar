# ── GAME LAW ──────────────────────────────
# Hunting and fishing are regulated to
# prevent the extinction of key species.
# A minimal intervention — primarily
# protecting upper-class recreational
# hunting grounds.

law_game_law = {
	group = lawgroup_environmental_regulations
	icon = "gfx/interface/icons/law_icons/smd/game_law.dds"

	progressiveness = 25

	modifier = {
		# Marginal pollution reduction from limited conservation
		state_pollution_generation_add              = -250
		# Small agricultural throughput cost from land restrictions
		building_group_bg_agriculture_throughput_add = -0.01
		interest_group_ig_intelligentsia_pol_str_mult      = 0.05
		interest_group_ig_rural_folk_pol_str_mult      = -0.10
	}

	ai_will_do = {
		NOR = {
			has_law = law_type:law_national_recycling
			has_law = law_type:law_protection_agency
			has_law = law_type:law_guaranteed_rights_of_nature
		}
	}

	ai_enact_weight_modifier = {
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		if = {
			limit = { is_great_power = yes }
			add = 50
		}
	}
}

# ── NATIONAL RECYCLING ────────────────────
# A national recycling programme reduces
# waste and resource consumption. Modest
# pollution reduction and small efficiency
# gains from recovered materials.

law_national_recycling = {
	group = lawgroup_environmental_regulations
	icon = "gfx/interface/icons/law_icons/smd/recycling.dds"

	progressiveness = 50

	modifier = {
		state_pollution_generation_add              = -750
		# Recovered materials reduce some input costs
		country_construction_goods_cost_mult        = -0.02
		# Small mortality improvement from cleaner waste management
		state_mortality_mult                        = -0.01
		interest_group_ig_industrialists_pol_str_mult      = 0.05
	}

	ai_will_do = {
		NOR = {
			has_law = law_type:law_protection_agency
			has_law = law_type:law_guaranteed_rights_of_nature
		}
		has_law_or_variant = law_type:law_game_law
	}

	ai_enact_weight_modifier = {
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		if = {
			limit = {
				has_law = law_type:law_game_law
				literacy_rate >= 0.40
			}
			add = 50
		}
	}
}

# ── PROTECTION AGENCY ─────────────────────
# A dedicated government agency enforces
# environmental standards — emissions limits,
# land use restrictions, industrial waste
# regulation. Unlocks institution_environmental_protection.

law_protection_agency = {
	group = lawgroup_environmental_regulations
	icon = "gfx/interface/icons/law_icons/smd/protection_agency.dds"

	progressiveness = 75

	modifier = {
		# Base pollution reduction from regulations
		state_pollution_generation_add              = -1000
		# Compliance costs reduce throughput modestly
		building_throughput_add                     = -0.02
		country_expenses_add                        = 20
		# Mortality improves from cleaner air and water
		state_mortality_mult                        = -0.02
		state_pollution_reduction_health_mult       = -0.10
		# Institution cost reduction
		country_institution_cost_institution_environmental_protection_mult = -0.20
	}

	institution = institution_environmental_protection

	ai_will_do = {
		NOT = { has_law = law_type:law_guaranteed_rights_of_nature }
		has_law_or_variant = law_type:law_national_recycling
	}

	ai_enact_weight_modifier = {
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		if = {
			limit = {
				has_law = law_type:law_national_recycling
				has_law = law_type:law_elected_bureaucrats
				literacy_rate >= 0.50
			}
			add = 75
		}
	}
}

# ── GUARANTEED RIGHTS OF NATURE ───────────
# Nature is granted constitutional or legal
# rights — rivers, forests, and ecosystems
# have standing in law. The most aggressive
# environmental framework possible.
# Significantly stronger institution effects.

law_guaranteed_rights_of_nature = {
	group = lawgroup_environmental_regulations
	icon = "gfx/interface/icons/law_icons/smd/rights_of_nature.dds"

	progressiveness = 100

	modifier = {
		# Strong base pollution reduction
		state_pollution_generation_add              = -2000
		# Significant industrial constraint
		building_throughput_add                     = -0.05
		building_group_bg_mining_throughput_add     = -0.05
		building_group_bg_agriculture_throughput_add = -0.02
		country_expenses_add                        = 40
		# Strong mortality and health improvements
		state_mortality_mult                        = -0.04
		state_pollution_reduction_health_mult       = -0.20
		# Food security improves from ecosystem protection
		state_food_security_add                     = 0.03
		# Institution cost reduction
		country_institution_cost_institution_environmental_protection_mult = -0.40
		country_institution_environmental_protection_max_investment_add   = 2
	}

	institution = institution_environmental_protection
	institution_modifier = {
		# Stronger per-level reduction than Protection Agency
		# -2500 per level (base -1500 from institution + -1000 here)
		state_pollution_generation_add              = -1000
		state_mortality_mult                        = -0.02
		state_food_security_add                     = 0.02
	}

	ai_will_do = {
		has_law_or_variant = law_type:law_protection_agency
		OR = {
			has_law = law_type:law_universal_suffrage
			has_law = law_type:law_census_voting
		}
	}

	ai_enact_weight_modifier = {
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		if = {
			limit = {
				has_law = law_type:law_protection_agency
				has_law = law_type:law_universal_suffrage
				literacy_rate >= 0.65
			}
			add = 75
		}
	}
}
