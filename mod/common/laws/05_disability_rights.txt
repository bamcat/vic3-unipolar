# ==========================================
# DISABILITY RIGHTS

law_minimal_recognition = {
	group = lawgroup_disability_rights
	icon = "gfx/interface/icons/law_icons/smd/minimal_recognition.dds"

	progressiveness = 0

	modifier = {
		# Disabled workers are excluded from most employment
		# reducing effective workforce and overall productivity
		state_pop_qualifications_mult               = -0.02
		state_mortality_mult                        = 0.02
	}

	ai_will_do = {
		always = yes
	}

	ai_enact_weight_modifier = {
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}

# ── WORKPLACE ACCOMMODATIONS ──────────────
# Employers are required to make reasonable
# adjustments to allow disabled workers to
# participate in the workforce. Productivity
# gains from an expanded working population,
# at modest cost to employers.

law_workplace_accommodations = {
	group = lawgroup_disability_rights
	icon = "gfx/interface/icons/law_icons/smd/workplace_accomodations.dds"

	progressiveness = 33

	modifier = {
		# Disabled workers enter the workforce
		state_pop_qualifications_mult               = 0.02
		# Employers bear some adjustment costs
		building_throughput_add                     = -0.01
		# Modest mortality improvement from reduced exclusion
		state_mortality_mult                        = 0.01
	}

	ai_will_do = {
		NOR = {
			has_law = law_type:law_anti_ableist_architecture
			has_law = law_type:law_public_treatments
		}
	}

	ai_enact_weight_modifier = {
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		if = {
			limit = {
				OR = {
					has_law = law_type:law_universal_suffrage
					has_law = law_type:law_census_voting
				}
				has_law = law_type:law_elected_bureaucrats
			}
			add = 75
		}
	}
}

# ── ANTI-ABLEIST ARCHITECTURE ─────────────
# Public buildings, transport, and
# infrastructure are required to be
# accessible. Construction costs rise
# as existing buildings are retrofitted,
# but population participation expands
# and standard of living improves.

law_anti_ableist_architecture = {
	group = lawgroup_disability_rights
	icon = "gfx/interface/icons/law_icons/smd/anti_ableism.dds"

	progressiveness = 66

	modifier = {
		# Accessible infrastructure broadens workforce participation
		state_pop_qualifications_mult               = 0.04
		state_lower_strata_standard_of_living_add   = 0.25
		state_middle_strata_standard_of_living_add  = 0.10
		# Retrofit and ongoing compliance costs
		country_construction_goods_cost_mult        = 0.02
		# Mortality improves as infrastructure no longer excludes
		state_mortality_mult                        = -0.01
	}

	ai_will_do = {
		NOT = { has_law = law_type:law_public_treatments }
		has_law_or_variant = law_type:law_workplace_accommodations
	}

	ai_enact_weight_modifier = {
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		if = {
			limit = {
				has_law = law_type:law_elected_bureaucrats
				has_law = law_type:law_workplace_accommodations
				literacy_rate >= 0.40
			}
			add = 75
		}
	}
}

# ── PUBLIC TREATMENTS ─────────────────────
# The state funds medical treatments,
# assistive devices, therapy, and support
# services for disabled citizens. Significant
# mortality reduction and standard of living
# gains across the population, at substantial
# ongoing cost.

law_public_treatments = {
	group = lawgroup_disability_rights
	icon = "gfx/interface/icons/law_icons/smd/public_treatments.dds"

	progressiveness = 100

	modifier = {
		# Full workforce participation and human capital gains
		state_pop_qualifications_mult               = 0.06
		state_standard_of_living_add                = 0.50
		state_lower_strata_standard_of_living_add   = 0.50
		# Significant mortality reduction from active treatment
		state_mortality_mult                        = -0.03
		# Substantial ongoing state expenditure
		# Education access improves as barriers fall
		state_education_access_add                  = 0.02
		state_welfare_payments_mult = 0.10
	}

	ai_will_do = {
		has_law_or_variant = law_type:law_anti_ableist_architecture
		OR = {
			has_law = law_type:law_public_health_insurance
			has_law = law_type:law_private_health_insurance
		}
	}

	ai_enact_weight_modifier = {
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
		if = {
			limit = {
				has_law = law_type:law_anti_ableist_architecture
				OR = {
					has_law = law_type:law_public_health_insurance
					has_law = law_type:law_private_health_insurance
				}
				literacy_rate >= 0.60
			}
			add = 75
		}
	}
}
