
start_israel_palestine_peacetime_journal_entry_system = {
	if = {
		limit = {
			exists = c:ISR
			exists = c:PAL
			NOT = { has_global_variable = ip_peacetime_started }
			NOT = { has_global_variable = intifada_started }
			c:ISR ?= {
				NOT = { has_war_with = c:PAL }
			}
		}

		add_contextless_journal_entry = je_israel_palestine_peacetime
	}
}

restart_israel_palestine_peacetime_after_escalation = {
	if = {
		limit = {
			exists = c:ISR
			exists = c:PAL
			NOT = { has_global_variable = ip_peacetime_started }
			NOT = { has_global_variable = intifada_started }
			c:ISR ?= {
				NOT = { has_war_with = c:PAL }
			}
		}

		set_global_variable = { name = ip_peacetime_negotiations value = 5 }
		set_global_variable = { name = ip_peacetime_security_coordination value = 5 }
		set_global_variable = { name = ip_peacetime_international_pressure value = 20 }

		add_contextless_journal_entry = je_israel_palestine_peacetime

		# Start a restarted peace process at a tenser 55 instead of the default 35.
		# Since the progress bar starts at 35, add +20.
		ip_temperature_add_20 = yes
	}
}

yugo_monthly_pressure_drift = {
	if = {
		limit = { c:YUG ?= this }
		change_variable = { name = yugo_secession_pressure add = 1 }
		change_variable = { name = yugo_stability add = -0.5 }

		if = {
			limit = { infamy >= infamy_threshold:infamous }
			change_variable = { name = yugo_secession_pressure add = 1 }
		}
		if = {
			limit = { legitimacy < 50 }
			change_variable = { name = yugo_secession_pressure add = 1 }
			change_variable = { name = yugo_stability add = -1 }
		}
		if = {
			limit = { has_law = law_type:law_outlawed_dissent }
			change_variable = { name = yugo_secession_pressure add = 1 }
		}

		# Player-side stabilising factors
		if = {
			limit = { has_law = law_type:law_universal_suffrage }
			change_variable = { name = yugo_secession_pressure add = -1 }
			change_variable = { name = yugo_stability add = 0.5 }
		}
		if = {
			limit = { has_modifier = yugoslav_confederal_negotiations }
			change_variable = { name = yugo_secession_pressure add = -1 }
			change_variable = { name = yugo_stability add = 1 }
		}
		if = {
			limit = { has_modifier = peacekeeping_force_present }
			change_variable = { name = yugo_secession_pressure add = -0.5 }
		}

		# Hard clamps so the bar can't go off scale
		if = {
			limit = { var:yugo_secession_pressure > 100 }
			set_variable = { name = yugo_secession_pressure value = 100 }
		}
		if = {
			limit = { var:yugo_secession_pressure < 0 }
			set_variable = { name = yugo_secession_pressure value = 0 }
		}
		if = {
			limit = { var:yugo_stability > 100 }
			set_variable = { name = yugo_stability value = 100 }
		}
		if = {
			limit = { var:yugo_stability < 0 }
			set_variable = { name = yugo_stability value = 0 }
		}
	}
}

yugo_launch_croatian_war = {

		}
		c:HRV = {
			set_variable = croatian_war_started
			create_diplomatic_play = {
				name = croatian_war_of_independence

				war = yes
				add_war_goal = {
					holder = c:YUG
					type = annex_country
					target_country = c:HRV
				}
			}
		}
	}
}
