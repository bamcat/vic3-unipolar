
# These mirror the scripted GUI buttons but add global anti-spam cooldowns.
# The global cooldown prevents all AI countries from sending repeated applications
# to the same organization's member states.

io_ai_common_application_cooldowns = {
	set_variable = { name = io_ai_application_cooldown years = 3 }
}

io_ai_apply_to_un = {
	io_ai_common_application_cooldowns = yes
	set_variable = { name = io_pending_org value = 1 }
	set_variable = { name = io_cooldown_un years = 1 }
	set_global_variable = { name = io_ai_un_application_global_cooldown years = 1 }
	save_scope_as = io_applicant
	every_country = {
		limit = {
			is_target_in_global_variable_list = { name = io_un_members target = this }
			NOT = { this = ROOT }
		}
		trigger_event = { id = io_events.1 days = 0 popup = yes }
	}
	trigger_event = { id = io_events.2 days = 30 }
}
