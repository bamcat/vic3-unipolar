# Diplomatic Unification scripted effects

smd_create_economic_union_with_target = {
	save_scope_as = smd_economic_union_source
	scope:target_country = {
		save_scope_as = smd_economic_union_target
	}

	create_treaty = {
		name = treaty_name_smd_economic_union
		first_country = scope:smd_economic_union_source
		second_country = scope:smd_economic_union_target
		is_draft = no
		binding_period = { years = 25 }
		articles_to_create = {
			{
				article = trade_privilege
				source_country = scope:smd_economic_union_source
				target_country = scope:smd_economic_union_target
			}
			{
				article = trade_privilege
				source_country = scope:smd_economic_union_target
				target_country = scope:smd_economic_union_source
			}
			{
				article = transit_rights
				source_country = scope:smd_economic_union_source
				target_country = scope:smd_economic_union_target
			}
			{
				article = transit_rights
				source_country = scope:smd_economic_union_target
				target_country = scope:smd_economic_union_source
			}
		}
	}

	set_variable = {
		name = smd_economic_union_partner
		value = scope:smd_economic_union_target
	}

	scope:smd_economic_union_target = {
		set_variable = {
			name = smd_economic_union_partner
			value = scope:smd_economic_union_source
		}
	}

	change_relations = {
		country = scope:smd_economic_union_target
		value = 25
	}

	scope:smd_economic_union_target = {
		change_relations = {
			country = scope:smd_economic_union_source
			value = 25
		}
	}
}

smd_complete_diplomatic_unification_with_target = {
	save_scope_as = smd_unification_source
	scope:target_country = {
		save_scope_as = smd_unification_target
	}

	custom_tooltip = {
		text = smd_diplomatic_unification_annexation_effect_tt
	}

	change_relations = {
		country = scope:smd_unification_target
		value = 50
	}

	annex = scope:smd_unification_target
}
