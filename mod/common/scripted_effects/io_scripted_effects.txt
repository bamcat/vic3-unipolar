io_create_starting_nato_defensive_pacts = {
	every_country = {
		limit = {
			is_target_in_global_variable_list = { name = io_nato_members target = this }
		}

		save_scope_as = nato_country_1

		every_country = {
			limit = {
				is_target_in_global_variable_list = { name = io_nato_members target = this }
				NOT = { this = scope:nato_country_1 }

				# Don't create duplicate defensive pacts
				NOR = {
					any_scope_treaty = {
						binds = scope:nato_country_1
						any_scope_article = {
							has_type = defensive_pact
						}
					}
				}
			}

			save_scope_as = nato_country_2

			scope:nato_country_1 = {
				create_treaty = {
					name = io_nato_defensive_pact_name
					first_country = scope:nato_country_1
					second_country = scope:nato_country_2
					is_draft = no
					binding_period = { years = 999 }
					articles_to_create = {
						{
							article = defensive_pact
						}
					}
				}
			}
		}
	}
}

### ══════════════════════════════════════════
### DEFAULT TREATIES (1992)
### Most recognized countries are signatories
### Notable non-signatories handled by exclusion
### ══════════════════════════════════════════
 
io_setup_default_treaties = {
	# Geneva Conventions — virtually universal by 1992
	every_country = {
		limit = { is_country_type = recognized }
		set_variable = io_treaty_geneva_signed
		add_modifier = { name = io_treaty_geneva_signatory }
	}
 
	# NPT — near-universal, notable non-signatories: Israel, India, Pakistan
	every_country = {
		limit = {
			is_country_type = recognized
			NOT = { this = c:ISR }
			NOT = { this = c:IDA }
			NOT = { this = c:PAK }
		}
		set_variable = io_treaty_npt_signed
		add_modifier = { name = io_treaty_npt_signatory }
	}
 
	# BWC — widely signed by 1992, but not universal
	every_country = {
		limit = {
			is_country_type = recognized
		}
		set_variable = io_treaty_bwc_signed
		add_modifier = { name = io_treaty_bwc_signatory }
	}
 
	# CWC — opened for signature in Jan 1993, so NOT signed at 1992 start
	# Players can sign it during gameplay
 
	# Convention Against Torture — widespread but not universal by 1992
	# Major signatories: Western nations, some others
	every_country = {
		limit = {
			is_country_type = recognized
			OR = {
				# Europe
				any_scope_state = { is_in_geographic_region = geographic_region:geographic_region_europe }
				# Americas
				any_scope_state = { is_in_geographic_region = geographic_region:geographic_region_north_america }
				any_scope_state = { is_in_geographic_region = geographic_region:geographic_region_south_america }
				# Major Asian democracies
				this = c:JPN
				this = c:KOR
				this = c:AUL
				this = c:NZL
			}
		}
		set_variable = io_treaty_cat_signed
		add_modifier = { name = io_treaty_cat_signatory }
	}
}

### Called by the yearly audit
### ══════════════════════════════════════════
 
io_ngo_check_high_support = {
	# IAEA — 75+ opinion grants tech spread bonus
	if = {
		limit = {
			has_variable = io_ngo_iaea_opinion
			var:io_ngo_iaea_opinion >= 75
			NOT = { has_modifier = io_ngo_iaea_high_support }
		}
		add_modifier = { name = io_ngo_iaea_high_support }
	}
	if = {
		limit = {
			OR = {
				NOT = { has_variable = io_ngo_iaea_opinion }
				var:io_ngo_iaea_opinion < 75
			}
			has_modifier = io_ngo_iaea_high_support
		}
		remove_modifier = io_ngo_iaea_high_support

	# RED CROSS — 75+ opinion grants prestige bonus

		limit = {

			}
			has_modifier = io_ngo_redcross_high_support

			}

		}

	}
}

io_rejection_cleanup = {
	# If a country has io_application_rejected, remove them from
	# whichever list they were optimistically added to
	if = {
		limit = { has_variable = io_application_rejected }
 
		# Check each org list and remove if present
		if = {
			limit = { is_target_in_global_variable_list = { name = io_un_members target = this } }
			remove_list_global_variable = { name = io_un_members target = this }
		}
		if = {
			limit = { is_target_in_global_variable_list = { name = io_nato_members target = this } }
			remove_list_global_variable = { name = io_nato_members target = this }
			remove_modifier = io_military_defense_agreement_member

			every_scope_treaty = {
