# Great Polarization / 2ACW scripted effects

great_polarization_initialize = {
	if = {
		limit = { NOT = { has_variable = great_polarization_intensity } }
		set_variable = { name = great_polarization_intensity value = 35 }
	}
	if = {
		limit = { NOT = { has_variable = great_polarization_left_pressure } }
		set_variable = { name = great_polarization_left_pressure value = 0 }
	}
	if = {
		limit = { NOT = { has_variable = great_polarization_right_pressure } }
		set_variable = { name = great_polarization_right_pressure value = 0 }
	}
	if = {
		limit = { NOT = { has_variable = great_polarization_institutional_trust } }
		set_variable = { name = great_polarization_institutional_trust value = 65 }
	}

	add_modifier = {
		name = great_polarization_baseline
		days = -1
	}
}

great_polarization_update_presidential_modifiers = {
	remove_modifier = great_polarization_right_wing_president
	remove_modifier = great_polarization_left_wing_president

	if = {
		limit = { great_polarization_ruler_is_right_wing = yes }
		add_modifier = {
			name = great_polarization_right_wing_president
			days = 35
		}
		change_variable = { name = great_polarization_left_pressure add = 3 }
		change_variable = { name = great_polarization_intensity add = 2 }
	}
	else_if = {
		limit = { great_polarization_ruler_is_left_wing = yes }
		add_modifier = {
			name = great_polarization_left_wing_president
			days = 35
		}
		change_variable = { name = great_polarization_right_pressure add = 3 }
		change_variable = { name = great_polarization_intensity add = 2 }
	}
	else_if = {
		limit = { great_polarization_ruler_is_liberal_or_moderate = yes }
		change_variable = { name = great_polarization_intensity add = -1 }
		change_variable = { name = great_polarization_institutional_trust add = 1 }
	}
}

great_polarization_check_for_second_civil_war = {
	if = {
		limit = {
			c:USA ?= this
			has_journal_entry = je_the_great_polarization
			NOT = { has_global_variable = second_american_civil_war_started }
			great_polarization_civil_war_should_start = yes
		}

		set_global_variable = second_american_civil_war_started
		trigger_event = { id = great_polarization.100 popup = yes }
	}
}

great_polarization_begin_second_american_civil_war = {
	set_global_variable = second_american_civil_war_started

	add_modifier = {
		name = second_american_civil_war_outbreak
		days = 3650
	}

	add_journal_entry = {
		type = je_second_american_civil_war
	}

	great_polarization_create_2acw_breakaways = yes
	great_polarization_start_2acw_wars = yes
	great_polarization_offer_player_switch_2acw_side = yes
}

	create_country = { tag = HWI origin = c:USA province = p:xB5403E }
	
	create_country = { tag = NVO origin = c:USA province = p:xbd8033 }
	
	if = { limit = { exists = c:BAM } c:BAM = { set_variable = 2acw_blue_aligned } }
	if = { limit = { exists = c:REA } c:REA = { set_variable = 2acw_red_aligned } }

	if = {
		limit = { exists = c:TXS }
		c:TXS = {
			set_variable = 2acw_red_protectorate
			set_variable = 2acw_red_aligned
		}
	}
	if = {
		limit = { exists = c:CAL }
		c:CAL = {
			set_variable = 2acw_blue_protectorate
			set_variable = 2acw_blue_aligned
		}
	}

	if = { limit = { exists = c:HWI } c:HWI = { set_variable = 2acw_neutral } }
	if = { limit = { exists = c:ALK } c:ALK = { set_variable = 2acw_neutral } }
	if = { limit = { exists = c:NAF } c:NAF = { set_variable = 2acw_unaligned_negotiable } }
	if = { limit = { exists = c:NVO } c:NVO = { set_variable = 2acw_unaligned_negotiable } }

	great_polarization_transfer_2acw_territory = yes
	great_polarization_create_2acw_armies = yes
}

great_polarization_start_2acw_wars = {
	if = {
		limit = {
			exists = c:REA
			exists = c:BAM
			NOT = { c:REA = { has_war_with = c:BAM } }
		}

		c:REA = {
			create_diplomatic_play = {
				name = second_american_civil_war
				type = dp_annex_war
				target_country = c:BAM
				war = yes

				add_war_goal = {
					holder = c:REA
					type = annex_country
					target_country = c:BAM
					primary_demand = yes
				}
				add_war_goal = {
					holder = c:BAM
					type = annex_country
					target_country = c:REA
				}
			}
		}
	}
}

great_polarization_offer_player_switch_2acw_side = {
	if = {
		limit = {
			c:USA ?= this
			is_player = yes
			has_global_variable = second_american_civil_war_started
		}

		if = {
			limit = {
				OR = {
					great_polarization_ruler_is_left_wing = yes
					great_polarization_ruler_is_liberal_or_moderate = yes
				}
			}
			trigger_event = { id = great_polarization.130 popup = yes }
		}
		else_if = {
			limit = { great_polarization_ruler_is_right_wing = yes }
			trigger_event = { id = great_polarization.131 popup = yes }
		}
		else = {
			trigger_event = { id = great_polarization.130 popup = yes }
		}
	}
}

great_polarization_breakaway_join_blue_side = {
	set_variable = 2acw_blue_aligned
	remove_variable = 2acw_red_aligned
	remove_variable = 2acw_neutral

	if = {
		limit = { exists = c:BAM }
		c:BAM = {
			create_treaty = {
				name = 2acw_blue_alignment_treaty
				first_country = c:BAM
				second_country = ROOT
				is_draft = no
				binding_period = { years = 10 }
				articles_to_create = {
					{ article = defensive_pact }
				}
			}
		}
	}
}
