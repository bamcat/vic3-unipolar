je_collapse_of_yugoslavia = {
	icon = "gfx/interface/icons/event_icons/event_war.dds"
	group = je_group_foreign_policy

	is_shown_when_inactive = {
		c:YUG ?= this
		year >= 1992
		NOT = { has_variable = yugoslavia_has_collapsed }
		NOT = { has_variable = yugoslavia_salvaged }
	}

	possible = {
		c:YUG ?= this
		year >= 1992
		NOT = { has_variable = yugoslavia_has_collapsed }
		NOT = { has_variable = yugoslavia_salvaged }
	}

	immediate = {
		if = { limit = { NOT = { has_variable = yugo_stability } } set_variable = { name = yugo_stability value = 50 } }
		if = { limit = { NOT = { has_variable = yugo_secession_pressure } } set_variable = { name = yugo_secession_pressure value = 50 } }
		if = { limit = { NOT = { has_variable = yugo_international_pressure } } set_variable = { name = yugo_international_pressure value = 0 } }
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_yugo_status_salvage_possible
				trigger = {
					var:yugo_stability >= 70
					var:yugo_secession_pressure <= 30
				}
			}
			triggered_desc = {
				desc = je_yugo_status_collapse_near
				trigger = {
					var:yugo_secession_pressure >= 75
				}
			}
			triggered_desc = {
				desc = je_yugo_status_unstable
				trigger = { always = yes }
			}
		}
	}

	complete = {
		OR = {
			custom_tooltip = {
				text = yugo_collapse_complete_collapsed_tt
				has_variable = yugoslavia_has_collapsed
			}
			AND = {
				custom_tooltip = {
					text = yugo_collapse_complete_salvaged_flag_tt
					has_variable = yugoslavia_salvaged
				}
				custom_tooltip = {
					text = yugo_collapse_complete_stability_tt
					var:yugo_stability >= 70
				}
				custom_tooltip = {
					text = yugo_collapse_complete_pressure_tt
					var:yugo_secession_pressure <= 30
				}
			}
		}
	}

	on_complete = {
		if = {
			limit = { has_variable = yugoslavia_salvaged }
			add_modifier = {
				name = yugoslav_confederal_compromise
				days = 1825
			}
			custom_tooltip = yugoslavia_salvaged_tt
		}
		else = {
			set_variable = yugoslavia_has_collapsed
			remove_variable = yugoslavia_tensions_rising
			add_modifier = {
				name = post_yugoslav_turmoil
				days = 1825
			}
			custom_tooltip = yugoslavia_collapse_tt
		}
	}

	on_monthly_pulse = {
		effect = {
			yugo_monthly_pressure_drift = yes

			if = {
				limit = {
					NOT = { has_variable = yugoslav_collapse_started }
					var:yugo_secession_pressure >= 80
					game_date >= 1992.1.1
				}
				trigger_event = { id = yugoslav_collapse.1 popup = yes }
			}

			if = {
				limit = {
					NOT = { has_variable = yugoslavia_salvaged }
					NOT = { has_variable = yugoslavia_has_collapsed }
					var:yugo_stability >= 70
					var:yugo_secession_pressure <= 30
				}
				trigger_event = { id = yugoslav_collapse.20 popup = yes }
			}
		}
	}

	should_be_pinned_by_default_uninvolved_or_context = yes
}

je_reunify_yugoslavia = {
	icon = "gfx/interface/icons/event_icons/event_war.dds"
	group = je_group_dilemmas

	is_shown_when_inactive = {
		c:YUG ?= this
		has_variable = yugoslavia_has_collapsed
	}

	possible = {
		c:YUG ?= this
		has_variable = yugoslavia_has_collapsed
	}

	complete = {
		OR = {
			custom_tooltip = {
				text = yugo_reunify_complete_flag_tt
				has_variable = yugoslavia_reunified
			}
			AND = {
				custom_tooltip = {
					text = yugo_reunify_state_bosnia_tt
					s:STATE_BOSNIA = { owner = c:YUG }
				}
				custom_tooltip = {
					text = yugo_reunify_state_croatia_tt
					s:STATE_CROATIA = { owner = c:YUG }
				}
				custom_tooltip = {
					text = yugo_reunify_state_slovenia_tt
					s:STATE_SLOVENIA = { owner = c:YUG }
				}
				custom_tooltip = {
					text = yugo_reunify_state_macedonia_tt
					s:STATE_MACEDONIA = { owner = c:YUG }
				}
				custom_tooltip = {
					text = yugo_reunify_state_montenegro_tt
					s:STATE_MONTENEGRO = { owner = c:YUG }
				}
				custom_tooltip = {
					text = yugo_reunify_state_southern_serbia_tt
					s:STATE_SOUTHERN_SERBIA = { owner = c:YUG }
				}
				custom_tooltip = {
					text = yugo_reunify_state_vlasenica_tt
					s:STATE_VLASENICA = { owner = c:YUG }
				}
			}
		}
	}

	fail = { game_date >= 2006.1.2 }

	on_complete = {
		set_variable = yugoslavia_reunified
		add_modifier = {
			name = yugoslav_restoration_success
			days = 1825
		}
		custom_tooltip = reunification_success_tt
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					has_diplomatic_pact = { country = c:HRV type = alliance }
					has_diplomatic_pact = { country = c:BOS type = alliance }
					num_subjects >= 3
				}
				set_variable = yugoslavia_reunified
			}
		}
	}
}

je_yugoslav_wars_serbia = {
	icon = "gfx/interface/icons/event_icons/event_war.dds"
	group = je_group_dilemmas

	is_shown_when_inactive = {
		OR = {
			c:YUG ?= this
			c:SRP ?= this
			c:HRV ?= this
			c:BOS ?= this
			c:MCD ?= this
			c:ALB ?= this
			c:KOS ?= this
			c:MNE ?= this
		}
	}

	possible = {
		OR = {
			c:YUG ?= this
			c:SRP ?= this
			c:HRV ?= this
			c:BOS ?= this
			c:MCD ?= this
			c:ALB ?= this
			c:KOS ?= this
			c:MNE ?= this
		}
		game_date < 2006.1.1
	}

	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_yugo_wars_status_kosovo
				trigger = { has_global_variable = kosovo_conflict_active }
			}
			triggered_desc = {
				desc = je_yugo_wars_status_bosnia
				trigger = { exists = c:BOS }
			}
			triggered_desc = {
				desc = je_yugo_wars_status_general
				trigger = { always = yes }
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = yugo_wars_complete_date_tt
			game_date >= 2006.1.1
		}
	}

	should_be_pinned_by_default_uninvolved_or_context = yes
}
