je_uranium_enrichment = {
	icon = "gfx/interface/icons/event_icons/nuclear_bomb.dds"
	group = je_group_domestic_policy
 
	should_be_pinned_by_default_uninvolved_or_context = yes
 
	is_shown_when_inactive = {
		has_variable = nuclear_enrichment_active
		NOT = { has_variable = nuclear_capable }
	}
 
	possible = {
		has_variable = nuclear_enrichment_active
	}
 
	on_monthly_pulse = {
		effect = {
			# Base advancement
			change_variable = { name = uranium_enrichment_progress add = 2 }
 
			# Extra per nuclear plant above 3
			if = {
				limit = {
					any_scope_state = {
						is_incorporated = yes
						has_building = building_nuclear_plant
						count >= 4
					}
				}
				change_variable = { name = uranium_enrichment_progress add = 1 }
			}
			if = {
				limit = {
					any_scope_state = {
						is_incorporated = yes
						has_building = building_nuclear_plant
						count >= 6
					}
				}
				change_variable = { name = uranium_enrichment_progress add = 1 }
			}
 
			# War slows enrichment
			if = {
				limit = { is_at_war = yes }
				change_variable = { name = uranium_enrichment_progress add = -1 }
			}
 
			# Clamp to 100
			if = {
				limit = { var:uranium_enrichment_progress >= 100 }
				set_variable = { name = uranium_enrichment_progress value = 100 }
			}
		}
 
		# Random events fire during enrichment
		random_events = {
			10 = 0
			2 = nuclear_events.2
		}
	}
 
	complete = {
		var:uranium_enrichment_progress >= 100
	}
 
	on_complete = {
		remove_variable = nuclear_enrichment_active
		set_variable = nuclear_capable
		trigger_event = { id = nuclear_events.1 popup = yes }
	}
 
	invalid = {
		NOT = { has_variable = nuclear_enrichment_active }
	}
 
	weight = 8000
	transferable = no
	can_revolution_inherit = no
}
 
# ── NUCLEAR PRODUCTION JE ────────────────
# Opens when player clicks Manufacture Warhead.
# Faster than enrichment — base 5/month.
# At 100: +1 to nukes variable.
 
je_nuclear_production = {
	icon = "gfx/interface/icons/event_icons/nuclear_bomb.dds"
	group = je_group_domestic_policy
 
	should_be_pinned_by_default_uninvolved_or_context = yes
 
	is_shown_when_inactive = {
		has_variable = nuclear_production_active
	}
 
	possible = {
		has_variable = nuclear_production_active
	}
 
	# scripted_progress_bar = nuclear_production_progress_bar
 
	on_monthly_pulse = {
		effect = {
			# Faster than enrichment — production is straightforward
			change_variable = { name = nuclear_production_progress add = 5 }
 
			# Each silo above 1 adds speed
			if = {
				limit = {
					any_scope_state = {
						is_incorporated = yes
						has_building = building_nuclear_silo
						count >= 2
					}
				}
				change_variable = { name = nuclear_production_progress add = 2 }
			}
 
			# Clamp
			if = {
				limit = { var:nuclear_production_progress >= 100 }
				set_variable = { name = nuclear_production_progress value = 100 }
			}
		}
	}
 
	complete = {
		var:nuclear_production_progress >= 100
	}
 
	on_complete = {
		remove_variable = nuclear_production_active
		# Increment warhead count
		if = {
			limit = { NOT = { has_variable = nukes } }
			set_variable = { name = nukes value = 1 }
		}
		else = {
			change_variable = { name = nukes add = 1 }
		}
		set_variable = { name = nuclear_production_progress value = 0 }
		trigger_event = { id = nuclear_events.3 popup = yes }
	}
 
	invalid = {
		NOT = { has_variable = nuclear_production_active }
	}
 
	weight = 8001
	transferable = no
	can_revolution_inherit = no
}

		effect = {
			# Base 3/month — slower than warhead production
			# Requires active research investment
			change_variable = { name = icbm_progress add = 3 }
 
			# Literacy and schools accelerate development
			if = {
				limit = {
					literacy_rate >= 0.85

						value >= 5
					}
				}
				change_variable = { name = icbm_progress add = 2 }
			}
 
			if = {
				limit = { var:icbm_progress >= 100 }
				set_variable = { name = icbm_progress value = 100 }
			}
		}
	}
 
	complete = {
		var:icbm_progress >= 100
	}
 
	on_complete = {
		set_variable = icbm_programme_complete
		trigger_event = { id = nuclear_events.11 popup = yes }
	}
 
	invalid = {
		NOT = { has_variable = nuclear_capable }
	}
 
	weight = 7999
	transferable = no
	can_revolution_inherit = no
}
