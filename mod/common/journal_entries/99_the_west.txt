# ── SG 6: THE UNIPOLAR MOMENT ─────────────
# The final goal. China and Russia must
# either embrace democracy and align with
# the West, or be reduced below great
# power status.
 
je_west_endgame = {
	should_be_pinned_by_default_uninvolved_or_context = yes
	icon = "gfx/interface/icons/objectives/the_west.dds"
 
	group = je_group_objectives
 
	complete = {
		# Russia condition
		OR = {
			# Russia democratized and friendly
			AND = {
				exists = c:RUS
				NOT = { any_rival_country = { c:RUS ?= this } }
				c:RUS = {
					OR = {
						has_law = law_type:law_presidential_republic
						has_law = law_type:law_parliamentary_republic
						has_law = law_type:law_council_republic
					}
					has_law = law_type:law_universal_suffrage
				}
			}
			# Russia no longer a great power
			AND = {
				exists = c:RUS
				c:RUS = {
					country_rank < rank_value:great_power
				}
			}
			AND = {
				exists = c:RUS
				c:RUS = {
					"relations(root.owner)" >= relations_threshold:cordial
				}
			}
			# Russia does not exist
			NOT = { exists = c:RUS }
		}
		# China condition
		OR = {
			# China democratized and friendly
			AND = {
				exists = c:PRC
				NOT = { any_rival_country = { c:PRC ?= this } }
				c:PRC = {
					OR = {
						has_law = law_type:law_presidential_republic
						has_law = law_type:law_parliamentary_republic
						has_law = law_type:law_council_republic
					}
					has_law = law_type:law_universal_suffrage
				}
			}
			# China no longer a great power
			AND = {
				exists = c:PRC
				c:PRC = {
					country_rank < rank_value:great_power
				}
			}
			AND = {
				exists = c:PRC
				c:PRC = {
					"relations(root.owner)" >= relations_threshold:cordial
				}
			}
			# China does not exist
			NOT = { exists = c:PRC }
		}
	}
 
	is_progressing = {
		OR = {
			# Either rival is already resolved
			AND = {
				exists = c:RUS
				c:RUS = { country_rank < rank_value:major_power }
			}
			AND = {
				exists = c:PRC
				c:PRC = { country_rank < rank_value:major_power }
			}
			AND = {
				exists = c:RUS
				c:RUS = {
					OR = {
						has_law = law_type:law_presidential_republic
						has_law = law_type:law_parliamentary_republic
					}
				}
			}
			AND = {
				exists = c:PRC
				c:PRC = {
					OR = {
						has_law = law_type:law_presidential_republic
						has_law = law_type:law_parliamentary_republic
					}
				}
			}
		}
	}
 
	weight = 2005
	transferable = yes
}
