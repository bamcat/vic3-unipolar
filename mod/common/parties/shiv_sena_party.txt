# Fallback party for Devout
# When devout are powerful, other religious IG's might join

REPLACE_OR_CREATE:shiv_sena_party = {
	color = { 230 100 25 }
	name = {
		first_valid = {
			triggered_desc = {
				desc = party_shiv_sena
				trigger = {
					always = yes
				}
			}
		}
	}
	icon = {
		default = "/gfx/interface/icons/political_parties/shiv_sena.dds"
	}
	
	# THIS IS RUN ONLY ONCE AT GAME START
	valid_for_country = {
		OR = { 
			THIS = c:IDA
			THIS = c:BHT
		}
	}

		available_for_interest_group = {
		is_marginal = no
		OR = {
			leader = { has_ideology = ideology:ideology_theocrat_leader }
			leader = { has_ideology = ideology:ideology_jihadist_leader }
			leader = { has_ideology = ideology:ideology_conservative_leader }
			leader = { has_ideology = ideology:ideology_monarchist_leader }
			leader = { has_ideology = ideology:ideology_corporatist_leader }
			leader = { has_ideology = ideology:ideology_right_wing_populist_leader }
			has_ideology = ideology:ideology_theocrat
			has_ideology = ideology:ideology_traditionalist
			has_ideology = ideology:ideology_conservative
			has_ideology = ideology:ideology_isolationist
			has_ideology = ideology:ideology_authoritarian
			has_ideology = ideology:ideology_monarchist
			is_interest_group_type = ig_devout
		}
	}

	# Root - IG considering joining
	# scope:number - the number of IGs that would join this party before the root IG
	join_weight = {
		value = {
			value = -5
			desc = "base_value"
		}
		if = {
			limit = {
				OR = {
				leader = { has_ideology = ideology:ideology_theocrat_leader }
				leader = { has_ideology = ideology:ideology_jihadist_leader }
			}
			}
			add = { value = 45 desc = "leader_ideological_compatibility" }
		}
		if = {
			limit = {
				OR = {
				leader = { has_ideology = ideology:ideology_conservative_leader }
				leader = { has_ideology = ideology:ideology_monarchist_leader }
				leader = { has_ideology = ideology:ideology_corporatist_leader }
				leader = { has_ideology = ideology:ideology_right_wing_populist_leader }
			}
			}
			add = { value = 20 desc = "leader_adjacent_ideology" }
		}
		if = {
			limit = {
				OR = {
				leader = { has_ideology = ideology:ideology_anarchist_leader }
				leader = { has_ideology = ideology:ideology_marxist_leader }
				leader = { has_ideology = ideology:ideology_social_liberal_leader }
				leader = { has_ideology = ideology:ideology_progressive_leader }
				leader = { has_ideology = ideology:ideology_green_leader }
				leader = { has_ideology = ideology:ideology_technocrat_leader }
			}
			}
			add = { value = -50 desc = "leader_ideological_incompatibility" }
		}
		if = {
			limit = {
				OR = {
				has_ideology = ideology:ideology_theocrat
				has_ideology = ideology:ideology_traditionalist
			}
			}
			add = { value = 35 desc = "interest_group_ideological_compatibility" }
		}
		if = {
			limit = {
				OR = {
				has_ideology = ideology:ideology_conservative
				has_ideology = ideology:ideology_isolationist
				has_ideology = ideology:ideology_authoritarian
				has_ideology = ideology:ideology_monarchist
			}
			}
			add = { value = 15 desc = "interest_group_adjacent_ideology" }
		}
		if = {
			limit = {
				OR = {
				has_ideology = ideology:ideology_progressive
				has_ideology = ideology:ideology_individualist
				has_ideology = ideology:ideology_communist
				has_ideology = ideology:ideology_globalist
			}
			}
			add = { value = -35 desc = "interest_group_ideological_incompatibility" }
		}
		if = {
			limit = {
				is_interest_group_type = ig_devout
			}
			add = { value = 20 desc = "from_archetype_party" }
		}
		if = {
			limit = {
				owner = { has_law = law_type:law_state_religion }
			}
			add = { value = 25 desc = "state_religion" }
		}
		if = {
			limit = {
				owner = { has_law = law_type:law_state_atheism }
			}
			add = { value = -60 desc = "state_atheism" }
		}
		if = {
			limit = {
				is_powerful = yes
				OR = {
					leader = { has_ideology = ideology:ideology_theocrat_leader }
					leader = { has_ideology = ideology:ideology_jihadist_leader }
					has_ideology = ideology:ideology_theocrat
					has_ideology = ideology:ideology_traditionalist
				}
			}
			add = { value = 20 desc = "powerful_ig" }
		}
		if = {
			limit = {
				scope:number = 1
				is_powerful = no
				is_marginal = no
			}
			multiply = { value = 2 desc = "from_influential_supporting_member" }
		}
		else_if = {
			limit = { scope:number = 2 }
			multiply = { value = 0.75 desc = "from_two_prior_members" }
		}
	}

}
