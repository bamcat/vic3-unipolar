# Fallback party for Rural Folk
# Occasionally used for an anti-serfdom alliance
# More popular in Nordic countries

REPLACE_OR_CREATE:agrarian_party = {
	color = { 64 173 59 }
	name = {
		first_valid = {
			triggered_desc = {
				desc = party_green
				trigger = {
					always = yes
				}
			}
		}
	}
	icon = {
		default = "/gfx/interface/icons/political_parties/agrarians.dds"
	}

	
	# THIS IS RUN ONLY ONCE AT GAME START
	valid_for_country = {
	NOT = {
		OR = {
			AND = {
				exists = c:GER
				THIS = c:GER
			}
		}
	}
	}

	available_for_interest_group = {
		is_marginal = no
		OR = {
			leader = { has_ideology = ideology:ideology_green_leader }
			leader = { has_ideology = ideology:ideology_left_wing_populist_leader }
			leader = { has_ideology = ideology:ideology_right_wing_populist_leader }
			leader = { has_ideology = ideology:ideology_conservative_leader }
			leader = { has_ideology = ideology:ideology_moderate_leader }
			leader = { has_ideology = ideology:ideology_democratic_socialist_leader }
			has_ideology = ideology:ideology_communitarian
			has_ideology = ideology:ideology_regionalist
			has_ideology = ideology:ideology_left_economy
			has_ideology = ideology:ideology_conservative
			has_ideology = ideology:ideology_traditionalist
			has_ideology = ideology:ideology_isolationist
			is_interest_group_type = ig_rural_folk
		}
	}

	# Root - IG considering joining
	# scope:number - the number of IGs that would join this party before the root IG
	join_weight = {
		value = {
			value = -5
			desc = "base_value"
		}
		if = {
			limit = {
				OR = {
				leader = { has_ideology = ideology:ideology_green_leader }
				leader = { has_ideology = ideology:ideology_left_wing_populist_leader }
				leader = { has_ideology = ideology:ideology_right_wing_populist_leader }
			}
			}
			add = { value = 45 desc = "leader_ideological_compatibility" }
		}
		if = {
			limit = {
				OR = {
				leader = { has_ideology = ideology:ideology_conservative_leader }
				leader = { has_ideology = ideology:ideology_moderate_leader }
				leader = { has_ideology = ideology:ideology_democratic_socialist_leader }
			}
			}
			add = { value = 20 desc = "leader_adjacent_ideology" }
		}
		if = {
			limit = {
				OR = {
				leader = { has_ideology = ideology:ideology_neoliberal_leader }
				leader = { has_ideology = ideology:ideology_authoritarian_capitalist_leader }
				leader = { has_ideology = ideology:ideology_fascist_leader }
				leader = { has_ideology = ideology:ideology_technocrat_leader }
			}
			}
			add = { value = -50 desc = "leader_ideological_incompatibility" }
		}
		if = {
			limit = {
				OR = {
				has_ideology = ideology:ideology_communitarian
				has_ideology = ideology:ideology_regionalist
			}
			}
			add = { value = 35 desc = "interest_group_ideological_compatibility" }
		}
		if = {
			limit = {
				OR = {
				has_ideology = ideology:ideology_left_economy
				has_ideology = ideology:ideology_conservative
				has_ideology = ideology:ideology_traditionalist
				has_ideology = ideology:ideology_isolationist
			}
			}
			add = { value = 15 desc = "interest_group_adjacent_ideology" }
		}
		if = {
			limit = {
				OR = {
				has_ideology = ideology:ideology_far_right_economy
				has_ideology = ideology:ideology_globalist
				has_ideology = ideology:ideology_totalitarian
				has_ideology = ideology:ideology_militarist
			}
			}
			add = { value = -35 desc = "interest_group_ideological_incompatibility" }
		}
		if = {
			limit = {
				is_interest_group_type = ig_rural_folk
			}
			add = { value = 20 desc = "from_archetype_party" }
		}
		if = {
			limit = {
				is_powerful = yes
				OR = {
					leader = { has_ideology = ideology:ideology_green_leader }
					leader = { has_ideology = ideology:ideology_left_wing_populist_leader }
					leader = { has_ideology = ideology:ideology_right_wing_populist_leader }
					has_ideology = ideology:ideology_communitarian
					has_ideology = ideology:ideology_regionalist
				}
			}
			add = { value = 20 desc = "powerful_ig" }
		}
		if = {
			limit = {
				scope:number = 1
				is_powerful = no
				is_marginal = no
			}
			multiply = { value = 2 desc = "from_influential_supporting_member" }
		}
		else_if = {
			limit = { scope:number = 2 }
			multiply = { value = 0.75 desc = "from_two_prior_members" }
		}
	}
}
