movement_communist = {
	category = movement_category_ideological
	
	ideology = ideology_communist_movement
	character_ideologies = {
		ideology_marxist_leader
		ideology_state_socialist_leader
		ideology_anarchist_leader
		ideology_market_socialist_leader
	}
	
	creation_trigger = {
		
	}	
	creation_weight = {
		value = 100
	}	
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}
	character_support_trigger = {
		OR = {
			has_ideology = ideology:ideology_marxist_leader
			has_ideology = ideology:ideology_state_socialist_leader
			has_ideology = ideology:ideology_anarchist_leader
			has_ideology = ideology:ideology_market_socialist_leader
			interest_group ?= {
				has_ideology = ideology:ideology_far_left_economy
				has_ideology = ideology:ideology_communist
			}
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_marxist_leader
					has_ideology = ideology:ideology_state_socialist_leader
					has_ideology = ideology:ideology_anarchist_leader
					has_ideology = ideology:ideology_market_socialist_leader
				}
			}
			multiply = { value = 5 }
		}
	}

	pop_support_trigger = {
		NOR = {
			is_pop_type = capitalists
			is_pop_type = aristocrats
			is_pop_type = slaves
		}
	}

	pop_support_factors = {
		movement_support_low_urbanization
		movement_support_high_literacy
		movement_support_below_expected_sol
		movement_support_radicalization
		movement_support_academics
		movement_support_laborers
		movement_support_machinists
		movement_support_peasants
		movement_support_rural
		movement_support_soldiers_conscripts
	}
	
	pop_support_weight = {	
		add = {
			desc = "URBAN_STATE"
			value = 10
			multiply = state.state_urbanization_rate
		}	
	
		if = {
			limit = {
				strata = middle
			}
			multiply = {
				value = 0.10
				desc = "MIDDLE_NO_ICON"
			}
			if = {
				limit = {
					is_pop_type = academics
				}
				add = {
					value = 3
					desc = "POP_ACADEMICS"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = laborers
				}
				add = {
					value = 15
					desc = "POP_LABORERS"
				}
			}		
			else_if = {
				limit = {
					is_pop_type = machinists
				}
				add = {
					value = 25
					desc = "POP_MACHINISTS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 6
					desc = "POP_CLERKS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = peasants 
				}
				add = {
					value = 3
					desc = "POP_PEASANTS"
				}
			}
			else = {
				add = {
					value = 6
					desc = "LOWER_NO_ICON"
				}
			}

			if = {
				limit = {
					AND = {
						is_pop_type = soldiers
						pop_employment_building = building_conscription_center
					}
				}
				multiply = {
					value = 2
					desc = "CONSCRIPTS"
				}
			}
		}

		if = {
			limit = {
				standard_of_living < state.average_expected_sol
				strata = lower
			}
			add = { # Increases likelihood of support if impoverished
				desc = "IS_BELOW_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}
		}
		else_if = {
			limit = {
				standard_of_living > state.average_expected_sol
				strata = lower
			}
			add = { # Decreases likelihood of support if well-off
				desc = "IS_ABOVE_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}

		multiply = { # Multiplied by 2x max
			desc = "POP_RADICALS"
			value = pop_radical_fraction
			add = 1
		}
	}
	
	revolution = {
		possible = {
			political_movement_support >= 0.05
		}
		
		weight = {
			value = 100
		}	
	}
	
	secession = {
		possible = {
			always = no
			political_movement_support >= 0.05
			scope:clout <= 0
		}
		
		weight = {
			value = 0
		}		
	}	
	
	additional_radicalism_factors = {
		owner = {
			if = {
				limit = { has_institution = institution_workplace_safety }
				
				add = {
					value = 3
					subtract = "institution_investment_level(institution_workplace_safety)"
					multiply = 0.05
					desc = "INSTITUTION_FUNDING_LEVEL_workplace_safety"
				}
			}
			else = {
				add = {
					value = 0.1
					desc = "NO_INSTITUTION_workplace_safety"
				}		
			}		
		}
	}
}

			else = {
				add = {
					value = 9
					desc = "MIDDLE_NO_ICON"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 9
					desc = "POP_CLERKS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = soldiers
				}
				add = {
					value = 6
					desc = "POP_SOLDIERS"
				}
			}
			else = {
				add = {
					value = 3
					desc = "LOWER_NO_ICON"
				}
			}
		}
		
		owner = {
			if = {
				limit = {
					number_of_claims > 0
				}

				add = {
					value = 5 # Add a base value for having claims at all
					add = number_of_claims
					multiply = 5
					max = 50
					desc = "NUMBER_OF_CLAIMS"
				}
			}
		}

		if = {
			limit = {
				standard_of_living < state.average_expected_sol
				strata = middle
			}
			add = { # Increases likelihood of support if impoverished
				desc = "IS_BELOW_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}
		}
		else_if = {
			limit = {
				standard_of_living > state.average_expected_sol
				strata = middle
			}
			add = { # Decreases likelihood of support if well-off
				desc = "IS_ABOVE_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				min = -5
			}
		}

		if = {
			limit = {
				is_employed = no
			}
			multiply = { 
				value = 1.5
				desc = "UNEMPLOYED"
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}

		multiply = {
			value = 0.012
			multiply = pop_acceptance
			desc = "ACCEPTANCE"
		}
	}
	
	revolution = {
		possible = {
			political_movement_support >= 0.05
		}
		
		weight = {
			value = 100
		}	
	
		state_weight = {
			value = 1		
		}
	}
	
	secession = {
		possible = {
			always = no
			political_movement_support >= 0.05
			scope:clout <= 0
		}
		
		weight = {
			value = 0
		}		
	}	
	
	additional_radicalism_factors = {
		owner = {
			if = {
				limit = {
					owner.number_of_claims > 0
				}
				
				add = {
					value = 5 # Add a base value for having claims at all
					add = owner.number_of_claims
					multiply = 0.025
					max = 0.3
					desc = "NUMBER_OF_CLAIMS"
				}				
			}
		}
	}
