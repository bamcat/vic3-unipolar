# ── BEGIN ENRICHMENT ────────────────────────

# Button visibility: show only when no programme and not already capable
nuclear_can_begin_enrichment = {
	scope = country
	is_shown = {
		NOT = { has_variable = nuclear_enrichment_active }
		NOT = { has_variable = nuclear_capable }
		NOT = { has_variable = nuclear_production_active }
	}
	is_valid = {
		# Must have at least 3 nuclear plants
		# nuclear_plant_count_sv is a scripted value you define that counts
		# building_nuclear_plant in all states
		any_scope_state = {
			is_incorporated = yes
			has_building = building_nuclear_plant
			count >= 3
		}
		# Must be major or great power
		country_rank >= rank_value:major_power
		# Not already in default
		in_default = no
		NOT = { has_modifier = io_treaty_npt_signatory }
	}
	ai_is_valid = {
		country_rank >= rank_value:great_power
		NOT = { has_variable = nuclear_enrichment_active }
		NOT = { has_variable = nuclear_capable }
	}
	ai_chance = { value = 20 }
	ai_frequency = { months = 24 }
}

# The actual begin enrichment effect
nuclear_begin_enrichment = {
	scope = country
	is_shown = {
		NOT = { has_variable = nuclear_enrichment_active }
		NOT = { has_variable = nuclear_capable }
	}
	is_valid = {
		any_scope_state = {
			is_incorporated = yes
			has_building = building_nuclear_plant
			count >= 3
		}
		country_rank >= rank_value:major_power
		in_default = no
	}
	effect = {
		set_variable = nuclear_enrichment_active
		set_variable = {
			name = uranium_enrichment_progress
			value = 0
		}
		# Open the enrichment journal entry
		add_journal_entry = { type = je_uranium_enrichment }
	}
	ai_is_valid = { always = no }
}

# ── TARGET LIST — ADD STATE ──────────────────
# Adds a state to wmd_target_list.
# Also adds its owner to wmd_target_list_c if not present.
 
nuclear_add_target_state = {
	scope = country
	is_shown = {
		NOT = {
			any_in_list = {
				variable = wmd_target_list
				THIS = scope:target_list_selected_state
			}
		}
	}
	is_valid = {
		has_variable = nuclear_capable
		var:nukes >= 1
	}
	effect = {
		add_to_variable_list = {
			name = wmd_target_list
			target = scope:target_list_selected_state
		}
		# Add the owning country to wmd_target_list_c if not already present
		scope:target_list_selected_state = {
			owner = {
				if = {
					limit = {
						NOT = {
							any_in_list = {
								variable = wmd_target_list_c
								THIS = PREV
							}
						}
					}
					ROOT = {
						add_to_variable_list = {
							name = wmd_target_list_c
							target = PREV
						}
					}
				}
			}
		}
	}
	ai_is_valid = { always = no }
}
 
# ── TARGET LIST — REMOVE STATE ───────────────
# Removes a state from wmd_target_list.
# Also cleans up wmd_target_list_c if that country
# has no more states in the list.
 
nuclear_remove_target_state = {
	scope = country
	is_shown = {
		any_in_list = {
			variable = wmd_target_list
			THIS = scope:target_list_selected_state
		}
	}
	effect = {
		remove_list_variable = {
			name = wmd_target_list
			target = scope:target_list_selected_state
		}
		# Clean up country from wmd_target_list_c if no states remain
		every_in_list = {
			variable = wmd_target_list_c
			limit = {
				NOT = {
					any_in_list = {
						variable = wmd_target_list
						owner = THIS
					}
				}
			}
			ROOT = {
				remove_list_variable = {
					name = wmd_target_list_c
					target = PREV
				}
			}
		}
	}
	ai_is_valid = { always = no }
}

# ── STATE ELIGIBILITY CHECKS ────────────────
 
# Is this state a valid MISSILE target?
# Must be: enemy-owned AND (neighbouring player state OR
# bordering an ally OR bordering a military-base-host country)
# OR blockaded by the player
state_is_valid_missile_target = {
	scope = country
	is_shown = {
		OR = {
			# State borders a state owned by ROOT
			any_neighbouring_state = {
				owner = ROOT
			}
			# State borders a state owned by an ally
			any_neighbouring_state = {
				owner = {
					has_diplomatic_pact = {
						who = ROOT
						type = alliance
					}
				}
			}
			# State borders a country that hosts ROOT's military bases
			any_neighbouring_state = {
				owner = {
					any_scope_treaty = {
						any_scope_article = {
							has_type = host_military_bases
							source_country = ROOT
						}
					}
				}
			}
			# State is blockaded by ROOT
			is_blockaded_by = ROOT
			# ROOT has ICBMs — global range
			has_variable = icbm_programme_complete
		}
	}
}

conventional_fire_missiles = {
	scope = country
	is_shown = {
		has_variable = conventional_missiles
		var:conventional_missiles >= 1
		is_at_war = yes
		NOT = { has_variable = missile_cooldown }
	}
	is_valid = {
		has_variable = conventional_missiles
		var:conventional_missiles >= 1
		any_in_list = { variable = conventional_target_list always = yes }
		NOT = { has_variable = missile_cooldown }
	}
	effect = {
		# Initialize deployment % to 100 if never set
		if = {
			limit = { NOT = { has_variable = conv_deployment_percentage } }
			set_variable = { name = conv_deployment_percentage value = 100 }
		}
		# Strike handles missile consumption and building hits
		conventional_missile_strike = yes
		# Clear targets
		clear_variable_list = conventional_target_list
		clear_variable_list = conventional_target_list_c
		# 60 day cooldown
		set_variable = { name = missile_cooldown days = 60 }
	}
	ai_is_valid = { always = no }
}

# player has ICBMs

	scope = country
	is_shown = {
		OR = {
			has_variable = icbm_programme_complete

				owner = ROOT
			}
			any_scope_state = {
				any_neighbouring_state = {
					owner = ROOT
				}
			}
		}
	}
}

	scope = country
	is_shown = {
		is_at_war = yes

			is_army = yes
			any_combat_unit = {
				OR = {
					has_unit_type = unit_type:combat_unit_type_fixed_wing_aircraft
					has_unit_type = unit_type:combat_unit_type_multirole_aircraft
					has_unit_type = unit_type:combat_unit_type_stealth_aircraft
					has_unit_type = unit_type:combat_unit_type_uav
					has_unit_type = unit_type:combat_unit_type_autonomous_aircraft
				}
				percent >= 0.25
			}
		}
	}
}

# Produces 5 missiles per salvo.

produce_missiles = {
	scope = country
	is_shown = { always = yes }
	is_valid = {
		in_default = no
		country_rank >= rank_value:recognized
	}
	effect = {
		if = {
			limit = { NOT = { has_variable = conventional_missiles } }
			set_variable = { name = conventional_missiles value = 5 }
		}
		else = {
			change_variable = { name = conventional_missiles add = 5 }
		}
		# Small cost — representing production investment
		add_modifier = {
			name = missile_production_cost
			months = 3
		}
	}
	ai_is_valid = { always = no }
}
