# ==========================================
# un_events.201 — Hidden Result Processor
# Fires once, 120 days after proposal.
# Checks votes, notifies, applies modifier,
# cleans up all global and per-country vars.
# ==========================================
 
un_events.201 = {
	type = country_event
	hidden = yes
	placement = ROOT
 
	immediate = {
		if = {
			limit = {
				scripted_bar_progress = {
				 name = progress_bar_tag
				 value > 0.50
				}
			}
			# Save motion type locally on each country before cleanup destroys it
			every_country = {
				set_variable = { name = un_last_motion_type value = global_var:un_current_motion_type }
			}
			# Motion passed — notify and apply modifier to all countries
			every_country = {
				trigger_event = { id = un_events.300 days = 1 }
			}
		}
		else = {
			# Motion failed
			every_country = {
				trigger_event = { id = un_events.301 days = 1 }
			}
		}
 
		# ── CLEANUP ──────────────────────────────
		# Remove per-country vote flags
		every_country = {
			limit = { has_variable = un_voted_on_motion }
			remove_variable = un_voted_on_motion
			remove_variable = un_last_motion_type
			je:je_united_nations = { 
				set_bar_progress = {
				 value = 0
				 name = un_motion_progress_bar
				}
			}
		}
		# Reset global motion state
		remove_global_variable = un_motion_active
		set_global_variable = { name = un_motion_yes_votes value = 0 }
		set_global_variable = { name = un_motion_no_votes value = 0 }
	}
}

un_events.202 = {
	type = country_event
	hidden = yes
	placement = ROOT
 
	immediate = {
		if = {
			limit = {
				global_var:un_motion_yes_votes >= global_var:un_motion_no_votes
			}
			if = {
				limit = {
					global_var:un_current_motion_type = flag:motion_sanctions
					exists = global_var:un_sanction_target
				}
				# Save motion type locally before cleanup
				every_country = {
					set_variable = { name = un_last_motion_type value = global_var:un_current_motion_type }
				}
				global_var:un_sanction_target = {
					add_modifier = { name = un_sanctions_target months = 60 }
					trigger_event = { id = un_events.302 days = 1 }
				}
				every_country = {
					limit = { NOT = { this = global_var:un_sanction_target } }
					trigger_event = { id = un_events.304 days = 1 }
				}
			}
			else_if = {
				limit = {
					global_var:un_current_motion_type = flag:motion_condemnation
					exists = global_var:un_sanction_target
				}
				# Save motion type locally before cleanup
				every_country = {
					set_variable = { name = un_last_motion_type value = global_var:un_current_motion_type }
				}
				global_var:un_sanction_target = {
					add_modifier = { name = un_condemnation_target months = 60 }
					trigger_event = { id = un_events.303 days = 1 }
				}
				every_country = {
					limit = { NOT = { this = global_var:un_sanction_target } }
					trigger_event = { id = un_events.304 days = 1 }
				}
			}
		}
		else = {
			every_country = { trigger_event = { id = un_events.301 days = 1 } }
		}
 
		# Cleanup
		every_country = {
			limit = { has_variable = un_voted_on_motion }
			remove_variable = un_voted_on_motion
			remove_variable = un_last_motion_type
			je:je_united_nations = { 
				set_bar_progress = {
				 value = 0
				 name = un_motion_progress_bar
				}
			}
		}
		remove_global_variable = un_motion_active
		remove_global_variable = un_sanction_target
		set_global_variable = { name = un_motion_yes_votes value = 0 }
		set_global_variable = { name = un_motion_no_votes value = 0 }
	}
}

un_events.5 = {
	type = country_event
	title = un_events.5.t
	desc = un_events.5.d
	icon = "gfx/interface/icons/event_icons/event_united_nations.dds"
	event_image = { video = "europenorthamerica_before_the_battle" }
	duration = 1
 
	trigger = {
		NOT = { has_variable = already_got_participation_trophy }
		NOT = { has_war_with = scope:peacekeeper_requestor }
		NOT = { scope:peacekeeper_requestor ?= this }
		any_scope_character = {
				is_busy = no
				is_ruler = no
				has_role = general
			}
	}
 
	immediate = {
		set_variable = already_got_participation_trophy
		# Pre-save generals for options
		random_scope_character = {
			limit = {
				is_busy = no
				is_ruler = no
				has_role = general
			}
			save_scope_as = peacekeeper_general_1
		}
		random_scope_character = {
			limit = {
				is_busy = no
				is_ruler = no
				has_role = general
				NOT = { this = scope:peacekeeper_general_1 }
			}
			save_scope_as = peacekeeper_general_2
		}
	}
 
	# ── OPTION A: PARTICIPATE ─────────────
	# Send a general. Larger relations boost.
	# General is deployed for 60 days then
	# returns via un_events.5b.
 
	option = {
		name = un_events.5.a
		trigger = {
			any_scope_character = {
				is_busy = no
				is_ruler = no
				has_role = general
			}
		}
 
		# Choose first available general
		if = {
			limit = { exists = scope:peacekeeper_general_1 }
			scope:peacekeeper_general_1 = {
				set_variable = peacekeeper_general_var
				set_character_busy_and_immortal = yes
				ROOT = {
					set_variable = {
						name = peacekeeper_general_storage
						value = prev
					}
				}
			}
		}
		else_if = {
			limit = { exists = scope:peacekeeper_general_2 }
			scope:peacekeeper_general_2 = {
				set_variable = peacekeeper_general_var
				set_character_busy_and_immortal = yes
				ROOT = {
					set_variable = {
						name = peacekeeper_general_storage
						value = prev
					}
				}
			}
		}
 
		# Larger relations boost for full participation
		change_relations = {
			country = global_var:peacekeeper_requestor
			value = 25
		}
 
		# Also improve relations with other UNSC members
		every_country = {
			limit = {
				has_modifier = permanent_unsc_member
				NOT = { this = ROOT }
				NOT = { this = global_var:peacekeeper_requestor }
			}
			change_relations = {
				country = ROOT
				value = 5
			}
		}
 
		# Schedule return event in 60 days
		trigger_event = { id = un_events.6 days = 60 }
 
		ai_chance = {
			value = 25
			modifier = {
				trigger = {
					NOT = {
						any_scope_character = {
							is_busy = no
							is_ruler = no
							has_role = general
						}
					}
				}
				add = -60
			}
		}
	}
 
	# ── OPTION B: DECLINE ─────────────────
	# Do not participate. Relations hit.
 
	option = {
		name = un_events.5.b
		change_relations = {
			country = global_var:peacekeeper_requestor
			value = -5
		}
 
		ai_chance = { value = 75 }
	}
}
