# ── ndp_events.1 — Year 2: Choose focus ──
ndp_events.1 = {
	type = country_event
	placement = root

	event_image = { video = "unspecific_factory_closed" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	title = ndp_events.1.t
	desc = ndp_events.1.d
	flavor = ndp_events.1.f

	trigger = { has_variable = ndp_industrial NOT = { has_variable = ndpe1 } }
	immediate = { set_variable = ndpe1 }
	duration = 3

	# Option A: Heavy industry — iron, steel, coal
	option = {
		name = ndp_events.1.a
		set_variable = ndp_industrial_sub_choice
		add_modifier = {
			name = ndp_modifier_industrial_heavy
			months = 96
		}
		every_scope_state = {
			limit = {
				is_incorporated = yes
				any_scope_building = {
					is_building_type = building_iron_mine
					level >= 1
				}
			}
			add_modifier = {
				name = ndp_modifier_state_heavy_industry
				months = 96
			}
		}
		ai_chance = { base = 50 }
	}

	# Option B: Light manufacturing — textiles, consumer goods
	option = {
		name = ndp_events.1.b
		default_option = yes
		set_variable = ndp_industrial_sub_choice
		add_modifier = {
			name = ndp_modifier_industrial_light
			months = 96
		}
		every_scope_state = {
			limit = {
				is_incorporated = yes
				state_urbanization_rate >= 0.20
			}
			add_modifier = {
				name = ndp_modifier_state_light_industry
				months = 96
			}
		}
		ai_chance = { base = 50 }
	}
}
