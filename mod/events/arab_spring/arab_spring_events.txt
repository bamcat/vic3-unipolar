# ==========================================
# arab_spring.1 — The Spark
# Fired by on_action when a revolution
# including liberals/progressives occurs.
# Analogous to peoples_springtime.1
# ==========================================

arab_spring.18 = {
	type = country_event
	event_image = {
		video = "arab_spring"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	title = arab_spring.1.t
	desc = arab_spring.1.d
	
	flavor = arab_spring.1.f

	trigger = {
		game_date >= 2010.11.31
		is_in_geographic_region = geographic_region_middle_east
		NOT = {
			OR = {
				has_law = law:law_census_voting
				has_law = law:law_universal_suffrage
				c:ISR ?= this
			}
			has_law = law:law_elected_bureaucrats
		}
		NOT = { has_variable = completed_arab_spring }
		NOT = { 
			has_variable = asp_1_event
		}
		
	}

	immediate = {
		set_variable = asp_1_event
		set_global_variable = arab_spring_happening
		
		if = {
			limit = {
				NOT = {
					any_political_movement = {
						is_political_movement_type = movement_liberal
					}
				}
			}
			create_political_movement = { type = movement_liberal }
		}
		random_political_movement = {
			limit = { is_political_movement_type = movement_liberal }
			save_scope_as = protest_movement
		}
		if = {
			limit = {
				NOT = {
					any_political_movement = {
						is_political_movement_type = movement_traditionalist
					}
				}
				OR = {
					country_has_accepted_religion = sunni
					country_has_accepted_religion = shiite
				}
			}
			create_political_movement = { type = movement_traditionalist }
		}
		if = {
			limit = {
				scope:protest_movement ?= {
					any_supporting_character = { has_role = agitator }
				}
			}
			scope:protest_movement ?= {
				random_supporting_character = {
					limit = { has_role = agitator }
					save_scope_as = protest_ig_leader
				}
			}
		}
	}

	duration = 3

	option = {	# Rally behind the protests
		name = arab_spring.1.a
		add_radicals = { value = medium_radicals strata = lower }
		add_radicals = { value = small_radicals strata = middle }
		if = {
			limit = { exists = scope:protest_ig_leader }
			scope:protest_ig_leader = {
				add_modifier = {
					name = arab_spring_popular_dissident
					days = normal_modifier_time
				}
			}
		}
	}

	option = {	# Acknowledge grievances without endorsing street protests
		name = arab_spring.1.b
		default_option = yes
		add_radicals = { value = small_radicals strata = lower }
		if = {
			limit = { exists = scope:protest_ig_leader }
			scope:protest_ig_leader = {
				add_modifier = {
					name = arab_spring_regime_collaborator
					days = normal_modifier_time
				}
			}
		}
	}
}

# ==========================================
# arab_spring.8 — The Crackdown
# JE failure/timeout — uprising suppressed.
# Analogous to peoples_springtime.8
# ==========================================

arab_spring.8 = {
	type = country_event
	event_image = { video = "arab_spring" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	title = arab_spring.8.t
	desc = arab_spring.8.d
	flavor = arab_spring.8.f

	duration = 3

	trigger = {
		OR = {
			has_variable = arab_winter_revolution_crushed
			AND = {
				var:arab_spring_timer_var >= 12
				any_political_movement = {
					OR = {
						is_political_movement_type = movement_liberal
						is_political_movement_type = movement_progressive
					}
					political_movement_support <= 0.10
					is_being_suppressed = yes
				}
				NOT = {
					any_scope_character = {
						OR = {
							has_role = politician
							has_role = agitator
						}
						OR = {
							has_ideology = ideology:ideology_liberal_leader
							has_ideology = ideology:ideology_progressive_leader
						}
					}
				}
				NOT = { has_variable = completed_arab_spring }
			}
		}
	}

	immediate = {
		set_variable = { name = completed_arab_spring }
		if = {
			limit = { has_variable = arab_winter_revolution_crushed }
			remove_variable = arab_winter_revolution_crushed
		}
	}

	option = {	# Celebrate the restoration of order
		name = arab_spring.8.a
		add_modifier = {
			name = arab_winter_revolution_crushed
			days = normal_modifier_time
		}
	}

	option = {	# Quiet relief — no triumphalism
		name = arab_spring.8.b
		default_option = yes
		add_loyalists = { value = 0.1 }
	}

	after = {
		je:je_arab_spring ?= {
			remove_involved_country = ROOT
		}
		hidden_effect = {
			set_strategy = ai_strategy_conservative_agenda
		}
		# Restore the authoritarian laws saved at JE start
		if = {
			limit = {
				has_variable = authoritarian_distribution_of_power_law
				has_variable = authoritarian_free_speech_law
				has_variable = authoritarian_internal_security_law
			}
			if = {
				limit = {
					NOT = { has_law = var:authoritarian_distribution_of_power_law }
				}
				activate_law = var:authoritarian_distribution_of_power_law
			}
			if = {
				limit = {
					NOT = { has_law = var:authoritarian_free_speech_law }
				}
				activate_law = var:authoritarian_free_speech_law
			}
			if = {
				limit = {
					NOT = { has_law = var:authoritarian_internal_security_law }
				}
				activate_law = var:authoritarian_internal_security_law
			}
		}
	}
}

# Analogous to peoples_springtime.10

arab_spring.10 = {
	type = country_event
	event_image = { video = "opposition_speaker" }

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	title = arab_spring.10.t
	desc = arab_spring.10.d
	flavor = arab_spring.10.f

	duration = 3

	trigger = {
		AND = {
			OR = {
				has_variable = had_arab_spring_revolution
				any_interest_group = {
					is_in_government = yes
					OR = {
						has_ideology = ideology:ideology_liberal_movement
						leader = {
							OR = {
								has_ideology = ideology:ideology_liberal_leader
								has_ideology = ideology:ideology_progressive_leader
							}
						}
					}
				}
			}
			OR = {
				has_law_or_variant = law_type:law_census_voting
				has_law_or_variant = law_type:law_universal_suffrage
			}
		}
		NOT = { has_variable = completed_arab_spring }
	}

	immediate = {
		set_variable = completed_arab_spring
		# Tag the season for flavour text
		if = {
			limit = { OR = { month >= 11 month < 2 } }
			set_variable = arab_winter_season_winter
		}
		if = {
			limit = { month >= 2 month < 5 }
			set_variable = arab_winter_season_spring
		}
		if = {
			limit = { month >= 5 month < 8 }
			set_variable = arab_winter_season_summer
		}
		if = {
			limit = { month >= 8 month < 11 }
			set_variable = arab_winter_season_autumn
		}
		set_variable = { name = completed_arab_spring }
	}

	option = {	# Embrace the transition — push for consolidation
		name = arab_spring.10.a
		default_option = yes
		custom_tooltip = {
			text = arab_winter_tt
			set_variable = arab_winter
		}
	}

	option = {	# Prepare for the counter-revolution
		name = arab_spring.10.b
		custom_tooltip = {
			text = arab_winter_tt
			set_variable = arab_winter
		}

			text = arab_winter_counter_revolution_bonus_tt
			set_variable = { name = var_counter_revolution_bonus value = 1 }
		}
	}
}
