# ==========================================
# cartel.1 — Escobar Captured
# ==========================================

cartel.1 = {
	type = country_event
	placement = root

	event_image = { video = "middleeast_police_breaking_door" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	title = cartel.1.t
	desc = cartel.1.d
	flavor = cartel.1.f

	duration = 3

	trigger = {
		c:CLM ?= this
		has_journal_entry = je_drug_trafficking
		has_variable = cartel_crackdown
		NOT = { has_variable = escobar_captured }
		market_capital.market = {
			mg:opium = {
				market_goods_pricier > 0.25
			}
		}
	}

	immediate = {
		set_variable = escobar_captured
		# Check if turmoil exists for future Colombian Conflict trigger
		if = {
			limit = {
				any_scope_state = {
					is_incorporated = yes
					turmoil >= 0.25
				}
			}
			set_variable = colombian_conflict_trigger
		}
	}

	option = {
		name = cartel.1.a
		default_option = yes
		add_modifier = {
			name = cartel_escobar_captured
			months = 60
		}
		add_modifier = {
			name = cartel_crackdown_success
			months = 36
		}
		# Notify all Americas countries
		hidden_effect = {
			every_country = {
				limit = {
					is_in_geographic_region = geographic_region_north_america
					NOT = { this = root }
				}
				trigger_event = { id = cartel.10 days = 1 }
			}
			every_country = {
				limit = {
					is_in_geographic_region = geographic_region_south_america
					NOT = { this = root }
				}
				trigger_event = { id = cartel.10 days = 1 }
			}
		}
	}
}

	}

		remove_modifier = cartel_el_chapo_captured

		}
		# Sinaloa cartel resurgence in Sinaloa state if it exists
		if = {
			limit = {
				any_scope_state = {
					state_region = s:STATE_SINALOA
					owner = c:MEX
				}
			}
			every_scope_state = {
				limit = {
					state_region = s:STATE_SINALOA
					owner = c:MEX
				}
				add_modifier = {
					name = cartel_sinaloa_controlled
					months = 60
				}
			}
		}
	}
}

# ==========================================
# cartel.4 — Cartel Turf Division (MEX timeout)
# Fires when Mexico's JE times out without
# completing. Mexican states get cartel
# territory modifiers.
# ==========================================

cartel.4 = {
	type = country_event
	placement = root

	event_image = { video = "middleeast_battlefield_trenches" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	title = cartel.4.t
	desc = cartel.4.d
	flavor = cartel.4.f

	duration = 3

	trigger = { c:MEX ?= this }

	immediate = {
		set_variable = mexican_drug_war_trigger
		set_variable = mexican_drug_war_active
	}

	option = {
		name = cartel.4.a
		default_option = yes
		add_modifier = {
			name = cartel_narco_state
			months = 60
		}
		# Assign cartel territory modifiers to Mexican states
		# CJNG — central/western states
		every_scope_state = {
			limit = {
				owner = c:MEX
				OR = {
					state_region = s:STATE_JALISCO
					state_region = s:STATE_GUANAJUATO
					state_region = s:STATE_MICHOACAN
				}
			}
			add_modifier = {
				name = cartel_cjng_controlled
				months = 60
			}
		}
		# Gulf Cartel — northeast/Gulf coast
		every_scope_state = {
			limit = {
				owner = c:MEX
				OR = {
					state_region = s:STATE_TAMAULIPAS
					state_region = s:STATE_VERACRUZ
					state_region = s:STATE_NUEVO_LEON
				}
			}
			add_modifier = {
				name = cartel_gulf_controlled
				months = 60
			}
		}
		# Sinaloa Cartel — northwest
		every_scope_state = {
			limit = {
				owner = c:MEX
				OR = {
					state_region = s:STATE_SINALOA
					state_region = s:STATE_SONORA
					state_region = s:STATE_CHIHUAHUA
				}
			}
			add_modifier = {
				name = cartel_sinaloa_controlled
				months = 60
			}
		}
		# Northeast Cartel
		every_scope_state = {
			limit = {
				owner = c:MEX
				OR = {
					state_region = s:STATE_COAHUILA
					state_region = s:STATE_ZACATECAS
				}
			}
			add_modifier = {
				name = cartel_northeast_controlled
				months = 60
			}
		}
		add_journal_entry = { type = je_mexican_drug_war }
		hidden_effect = {
			set_variable = mexican_drug_war_active
		}
	}
}

			name = cartel_us_rejection

# cartel.8 — Cartel Retaliates Against Crackdown

	type = country_event
	placement = root

	event_image = { video = "bombed_city" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	title = cartel.8.t
	desc = cartel.8.d
	flavor = cartel.8.f

	duration = 3

	trigger = {
		has_journal_entry = je_drug_trafficking
		has_variable = cartel_crackdown
		NOT = { has_variable = cartel_retaliation_fired }
	}

	cooldown = { days = long_modifier_time }

	immediate = {
		set_variable = cartel_retaliation_fired
		random_scope_state = {
			limit = { is_incorporated = yes }
			save_scope_as = cartel_retaliation_state
		}
	}

	# Option A: Escalate — double down on the crackdown
	option = {
		name = cartel.8.a
		add_authority = -50
		add_modifier = {
			name = cartel_retaliation_escalate
			months = 18
		}
		scope:cartel_retaliation_state = {
			add_radicals_in_state = { value = medium_radicals }
		}
		ai_chance = { base = 50 }
	}

	# Option B: Pull back — protect civilians over progress
	option = {
		name = cartel.8.b
		default_option = yes
		add_modifier = {
			name = cartel_retaliation_pullback
			months = 12
		}
		ai_chance = { base = 50 }
	}
}
