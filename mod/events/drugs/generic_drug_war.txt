namespace = cartel_fight

# cartel_fight.0 — General Selection
cartel_fight.11 = {
	type = country_event
	placement = root

	event_image = { video = "unspecific_military_parade" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	title = cartel_fight.0.t
	desc = cartel_fight.0.d
	flavor = cartel_fight.0.f

	duration = 3

	trigger = {
		any_scope_character = {
			is_busy = no
			is_ruler = no
			has_role = general
		}
	}

	immediate = {
		random_scope_character = {
			limit = { is_busy = no is_ruler = no has_role = general }
			save_scope_as = cartel_general_1
		}
		random_scope_character = {
			limit = {
				is_busy = no is_ruler = no has_role = general
				NOT = { this = scope:cartel_general_1 }
			}
			save_scope_as = cartel_general_2
		}
		random_scope_character = {
			limit = {
				is_busy = no is_ruler = no has_role = general
				NOR = {
					this = scope:cartel_general_1
					this = scope:cartel_general_2
				}
			}
			save_scope_as = cartel_general_3
		}
	}

	option = {
		name = cartel_fight.0.a
		default_option = yes
		scope:cartel_general_1 = {
			set_variable = cartel_general_var
			set_character_busy_and_immortal = yes
			ROOT = {
				set_variable = { name = cartel_general_storage value = prev }
			}
		}
		ai_chance = { base = 50 }
	}

	option = {
		name = cartel_fight.0.b
		trigger = { exists = scope:cartel_general_2 }
		scope:cartel_general_2 = {
			set_variable = cartel_general_var
			set_character_busy_and_immortal = yes
			ROOT = {
				set_variable = { name = cartel_general_storage value = prev }
			}
		}
		ai_chance = { base = 30 }
	}

	option = {
		name = cartel_fight.0.c
		trigger = { exists = scope:cartel_general_3 }
		scope:cartel_general_3 = {
			set_variable = cartel_general_var
			set_character_busy_and_immortal = yes
			ROOT = {
				set_variable = { name = cartel_general_storage value = prev }
			}
		}
		ai_chance = { base = 20 }
	}
}

# cartel_fight.1 — Cartel Bust
cartel_fight.1 = {
	type = country_event
	placement = root

	event_image = { video = "middleeast_police_breaking_door" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/cocaine.dds"

	title = cartel_fight.1.t
	desc = cartel_fight.1.d
	flavor = cartel_fight.1.f

	duration = 3

	trigger = {
		has_journal_entry = je_fighting_cartels
		any_scope_character = { has_variable = cartel_general_var }
	}

	cooldown = { days = normal_modifier_time }

	immediate = {
		random_scope_character = {
			limit = { has_variable = cartel_general_var }
			save_scope_as = cartel_general
		}
	}

	cancellation_trigger = {
		NOT = { has_journal_entry = je_fighting_cartels }
	}

	option = {
		name = cartel_fight.1.a
		default_option = yes
		je:je_fighting_cartels = {
			add_progress = { name = cartel_suppression_bar value = 10 }
		}
	}
}

# cartel_fight.2 — Cartel Ambushes Troops
cartel_fight.2 = {
	type = country_event
	placement = root

	event_image = { video = "gunner" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/cocaine.dds"

	title = cartel_fight.2.t
	desc = cartel_fight.2.d
	flavor = cartel_fight.2.f

	duration = 14

	trigger = {
		has_journal_entry = je_fighting_cartels
		any_scope_character = { has_variable = cartel_general_var }
		je:je_fighting_cartels = {
			"scripted_bar_progress(cartel_suppression_bar)" >= 20
		}
	}

	cooldown = { days = long_modifier_time }

	immediate = {
		random_scope_character = {
			limit = { has_variable = cartel_general_var }
			save_scope_as = cartel_general
		}
	}

	cancellation_trigger = {
		NOT = { has_journal_entry = je_fighting_cartels }
	}

	option = {
		name = cartel_fight.2.a
		default_option = yes
		je:je_fighting_cartels = {
			add_progress = { name = cartel_suppression_bar value = -5 }
		}
		add_modifier = {
			name = cartel_fight_ambush
			months = 12
		}
	}
}

# cartel_fight.10 — Cartels Suppressed
cartel_fight.10 = {
	type = country_event
	placement = root

	event_image = { video = "room_clearing" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/cocaine.dds"

	title = cartel_fight.10.t
	desc = cartel_fight.10.d
	flavor = cartel_fight.10.f

	duration = 3
	trigger = { }

	immediate = {
		random_scope_character = {
			limit = { has_variable = cartel_general_var }
			save_scope_as = cartel_general
		}
	}

	option = {
		name = cartel_fight.10.a
		default_option = yes
		scope:cartel_general ?= {
			set_character_busy_and_immortal = no
			remove_variable = cartel_general_var
		}
		remove_variable = cartel_general_storage
		add_modifier = {
			name = cartel_fight_victory
			months = 36
		}
		add_loyalists = { value = medium_radicals strata = lower }
		add_loyalists = { value = small_radicals strata = middle }
	}
}
