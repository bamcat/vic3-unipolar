namespace = abortion_laws


# ==========================================
# abortion_laws.1
# "Life Begins at Conception"
# Fires when enacting banned abortions.
# The devout IG frames the ban in
# absolute moral terms.
# checkpoint_advance
# ==========================================

abortion_laws.1 = {
	type = country_event
	placement = ROOT

	title = abortion_laws.1.t
	desc = abortion_laws.1.d
	flavor = abortion_laws.1.f

	category = enactment

	event_image = {
		video = "unspecific_politicians_arguing"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_election.dds"

	duration = 3

	trigger = {
		is_enacting_law = law_type:law_banned_abortions
		NOT = { has_modifier = abortion_law_moral_absolute }
		any_interest_group = {
			is_interest_group_type = ig_devout
			ig_counts_as_marginal = no
		}
	}

	cooldown = { days = normal_modifier_time }

	immediate = {
		currently_enacting_law = {
			save_scope_as = abortion_law_scope
		}
		random_interest_group = {
			limit = {
				is_interest_group_type = ig_devout
				ig_counts_as_marginal = no
			}
			save_scope_as = devout_ig
			leader = {
				save_scope_as = devout_leader
			}
		}
	}

	# Option A: Embrace the moral framing — life is sacred and the state must protect it
	option = {
		name = abortion_laws.1.a
		default_option = yes
		add_enactment_modifier = {
			name = abortion_law_moral_absolute
			# = 2
		}
		scope:devout_ig = {
			add_modifier = {
				name = abortion_law_devout_empowered
				days = normal_modifier_time
			}
		}
		ai_chance = { base = 65 }
	}

	# Option B: Frame it as demographic policy — keep religion out of the justification
	option = {
		name = abortion_laws.1.b
		add_enactment_modifier = {
			name = abortion_law_moral_absolute
		}
		add_enactment_modifier = {
			name = abortion_law_demographic_framing
		}
		scope:devout_ig = {
			add_modifier = {
				name = abortion_law_devout_sidelined
				days = normal_modifier_time
			}
		}
		ai_chance = { base = 35 }
	}
}
