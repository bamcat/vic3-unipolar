namespace = environment_laws

# ==========================================
# environment_laws.1
# "Industry Offers Voluntary Compliance"
# Advance event. Fires for any progressive
# environmental law. Industrialists propose
# self-regulation to pre-empt enforcement.
# ==========================================

environment_laws.1 = {
	type = country_event
	placement = ROOT

	title = environment_laws.1.t
	desc = environment_laws.1.d
	flavor = environment_laws.1.f

	category = enactment

	event_image = {
		video = "asia_factory_accident"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_election.dds"

	duration = 3

	trigger = {
		OR = {
			is_enacting_law = law_type:law_game_law
			is_enacting_law = law_type:law_national_recycling
			is_enacting_law = law_type:law_protection_agency
			is_enacting_law = law_type:law_guaranteed_rights_of_nature
		}
		NOT = { has_modifier = env_voluntary_compliance }
		any_interest_group = {
			is_interest_group_type = ig_industrialists
			ig_counts_as_marginal = no
		}
	}

	cooldown = { days = long_modifier_time }

	immediate = {
		currently_enacting_law = {
			save_scope_as = env_law_scope
		}
		random_interest_group = {
			limit = {
				is_interest_group_type = ig_industrialists
				ig_counts_as_marginal = no
			}
			save_scope_as = industry_ig
			leader = {
				save_scope_as = industry_leader
			}
		}
	}

	# Option A: Accept — build voluntary compliance into the framework
	option = {
		name = environment_laws.1.a
		default_option = yes
		add_enactment_modifier = {
			name = env_voluntary_compliance
			# = 2
		}
		ai_chance = {
			base = 50
			modifier = {
				trigger = { has_law = law_type:law_laissez_faire }
				add = 25
			}
		}
	}

	# Option B: Reject — mandatory enforcement only
	option = {
		name = environment_laws.1.b
		add_enactment_modifier = {
			name = env_voluntary_compliance
		}
		add_enactment_modifier = {
			name = env_mandatory_enforcement
			# = 2
		}
		ai_chance = {
			base = 50
			modifier = {
				trigger = { has_law = law_type:law_interventionism }
				add = 25
			}
		}
	}
}

# ==========================================
# environment_laws.6
# "Environmental Disaster Mid-Enactment"
# Stall event. Fires for any law.
# A pollution event or ecological collapse
# either galvanises support or
# reveals the state's inability to act.
# ==========================================

environment_laws.6 = {
	type = country_event
	placement = scope:disaster_state_scope

	title = environment_laws.6.t
	desc = environment_laws.6.d
	flavor = environment_laws.6.f

	category = enactment

	event_image = {
		video = "unspecific_devastation"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	duration = 3

	trigger = {
		OR = {
			is_enacting_law = law_type:law_game_law
			is_enacting_law = law_type:law_national_recycling
			is_enacting_law = law_type:law_protection_agency
			is_enacting_law = law_type:law_guaranteed_rights_of_nature
		}
		NOT = { has_modifier = env_disaster_response }
		any_scope_state = {
			is_incorporated = yes
			pollution_amount >= 1000
		}
	}

	cooldown = { days = long_modifier_time }

	immediate = {
		currently_enacting_law = {
			save_scope_as = env_law_scope
		}
		random_scope_state = {
			limit = {
				is_incorporated = yes
				pollution_amount >= 1000
			}
			save_scope_as = disaster_state_scope
		}
	}

	# Option A: Use the disaster to accelerate — demand urgent passage
	option = {
		name = environment_laws.6.a
		default_option = yes
		add_enactment_modifier = {
			name = env_disaster_response
			# = 2
		}
		scope:disaster_state_scope = {
			add_modifier = {
				name = env_disaster_state_modifier
				months = 24
			}
		}
		ai_chance = {
			base = 60
			modifier = {
				trigger = { has_law = law_type:law_universal_suffrage }
				add = 20
			}
		}
	}

	# Option B: Contain the story — don't let a disaster become a crisis
	option = {
		name = environment_laws.6.b
		add_enactment_modifier = {
			name = env_disaster_response
		}
		add_enactment_modifier = {
			name = env_disaster_suppressed
			# = -1
		}
		scope:disaster_state_scope = {
			add_modifier = {
				name = env_disaster_state_modifier
				months = 36
			}
		}
		ai_chance = { base = 40 }
	}
}
