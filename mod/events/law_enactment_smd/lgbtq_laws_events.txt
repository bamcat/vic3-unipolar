# ==========================================
# lgbtq_laws.1
# "The Church Demands Action"
# Fires when enacting active oppression.
# The devout IG frames the law in
# explicitly moral and religious terms.
# checkpoint_advance
# ==========================================

lgbtq_laws.1 = {
	type = country_event
	placement = ROOT

	title = lgbtq_laws.1.t
	desc = lgbtq_laws.1.d
	flavor = lgbtq_laws.1.f

	category = enactment

	event_image = {
		video = "unspecific_politicians_arguing"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_election.dds"

	duration = 3

	trigger = {
		is_enacting_law = law_type:law_active_oppression
		NOT = { has_modifier = lgbtq_law_religious_mandate }
		any_interest_group = {
			is_interest_group_type = ig_devout
			ig_counts_as_marginal = no
		}
	}

	cooldown = { days = normal_modifier_time }

	immediate = {
		currently_enacting_law = {
			save_scope_as = lgbtq_law_scope
		}
		random_interest_group = {
			limit = {
				is_interest_group_type = ig_devout
				ig_counts_as_marginal = no
			}
			save_scope_as = devout_ig
			leader = {
				save_scope_as = devout_leader
			}
		}
	}

	# Option A: Embrace the religious framing — this is a matter of moral law
	option = {
		name = lgbtq_laws.1.a
		default_option = yes
		add_enactment_modifier = {
			name = lgbtq_law_religious_mandate
			# = 2
		}
		scope:devout_ig = {
			add_modifier = {
				name = lgbtq_law_devout_empowered
				days = normal_modifier_time
			}
		}
		ai_chance = { base = 65 }
	}

	# Option B: Frame it as a matter of social order — keep religion out of the justification
	option = {
		name = lgbtq_laws.1.b
		add_enactment_modifier = {
			name = lgbtq_law_religious_mandate
		}
		scope:devout_ig = {
			add_modifier = {
				name = lgbtq_law_devout_sidelined
				days = normal_modifier_time
			}
		}
		ai_chance = { base = 35 }
	}
}
