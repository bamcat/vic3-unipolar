# ==========================================
# disability_laws.1
# "Employers Push Back"
# Fires when enacting workplace accommodations.
# Industrialists argue the compliance costs
# are unworkable.
# ==========================================

disability_laws.1 = {
	type = country_event
	placement = ROOT

	title = disability_laws.1.t
	desc = disability_laws.1.d
	flavor = disability_laws.1.f

	category = enactment

	event_image = {
		video = "unspecific_politicians_arguing"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_election.dds"

	duration = 3

	trigger = {
		is_enacting_law = law_type:law_workplace_accommodations
		NOT = { has_modifier = disability_employer_pressure }
		any_interest_group = {
			OR = {
				is_interest_group_type = ig_industrialists
				is_interest_group_type = ig_petty_bourgeoisie
			}
			ig_counts_as_marginal = no
			law_stance = {
				law = law_type:law_workplace_accommodations
				value < neutral
			}
		}
	}
	
	cooldown = { days = normal_modifier_time }

	immediate = {
		currently_enacting_law = {
			save_scope_as = disability_law_scope
		}
		random_interest_group = {
			limit = {
				OR = {
					is_interest_group_type = ig_industrialists
					is_interest_group_type = ig_petty_bourgeoisie
				}
				ig_counts_as_marginal = no
				law_stance = {
					law = law_type:law_workplace_accommodations
					value < neutral
				}
			}
			save_scope_as = employer_ig
			leader = {
				save_scope_as = employer_leader
			}
		}
	}

	# Option A: Hold firm — compliance is non-negotiable
	option = {
		name = disability_laws.1.a
		default_option = yes
		add_enactment_modifier = {
			name = disability_employer_pressure
		}
		ai_chance = {
			base = 60
			modifier = {
				trigger = { has_law = law_type:law_elected_bureaucrats }
				add = 20
			}
		}
	}

	# Option B: Offer a transition period — slower compliance
	option = {
		name = disability_laws.1.b
		add_enactment_modifier = {
			name = disability_employer_pressure
			# = 2
		}
		add_enactment_modifier = {
			name = disability_compliance_delay
			# = -1
		}
		ai_chance = { base = 40 }
	}
}
