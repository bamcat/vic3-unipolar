# ==========================================
# childrens_childcare.1
# "The Market Knows Best"
# Fires when enacting privatized preschool.
# Business interests back the law — but
# critics warn access will be unequal.
# ==========================================

childrens_childcare.1 = {
	type = country_event
	placement = ROOT

	title = childrens_childcare.1.t
	desc = childrens_childcare.1.d
	flavor = childrens_childcare.1.f

	category = enactment

	event_image = {
		video = "unspecific_politicians_arguing"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_election.dds"

	duration = 3

	trigger = {
		is_enacting_law = law_type:law_privatized_preschool_education
		NOT = { has_modifier = childrens_rights_market_preschool_framing }
		any_interest_group = {
			is_interest_group_type = ig_petty_bourgeoisie
			ig_counts_as_marginal = no
		}
	}

	cooldown = { days = normal_modifier_time }

	immediate = {
		currently_enacting_law = {
			save_scope_as = childrens_rights_law_scope
		}
		random_interest_group = {
			limit = {
				is_interest_group_type = ig_petty_bourgeoisie
				ig_counts_as_marginal = no
			}
			save_scope_as = business_ig
			leader = {
				save_scope_as = business_leader
			}
		}
	}

	# Option A: Let the market lead — competition will drive quality
	option = {
		name = childrens_childcare.1.a
		default_option = yes
		add_enactment_modifier = {
			name = childrens_rights_market_preschool_framing
			# = 2
		}
		scope:business_ig = {
			add_modifier = {
				name = childrens_rights_business_engaged
				days = normal_modifier_time
			}
		}
		ai_chance = {
			base = 60
			modifier = {
				trigger = {
					exists = ruler
					ruler = { has_ideology = ideology_market_liberal }
				}
				add = 20
			}
		}
	}

	# Option B: Regulate pricing — private providers must meet access standards
	option = {
		name = childrens_childcare.1.b
		add_enactment_modifier = {
			name = childrens_rights_market_preschool_framing
		}
		add_enactment_modifier = {
			name = childrens_rights_access_mandate
		}
		ai_chance = {
			base = 40
			modifier = {
				trigger = {
					exists = ruler
					ruler = {
						OR = {
							has_ideology = ideology_liberal_leader
							has_ideology = ideology_progressive_leader
						}
					}
				}
				add = 20
			}
		}
	}
}
