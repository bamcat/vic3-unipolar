namespace = gun_laws

# ==========================================
# gun_laws.1
# "The Safety Argument"
# Fires when enacting banned firearms.
# Public health advocates and the
# intelligentsia make the case that
# fewer guns means fewer deaths.
# checkpoint_advance
# ==========================================

gun_laws.1 = {
	type = country_event
	placement = ROOT

	title = gun_laws.1.t
	desc = gun_laws.1.d
	flavor = gun_laws.1.f

	category = enactment

	event_image = {
		video = "unspecific_politicians_arguing"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_election.dds"

	duration = 3

	trigger = {
		is_enacting_law = law_type:law_banned_firearms
		NOT = { has_modifier = gun_law_safety_argument }
		any_interest_group = {
			is_interest_group_type = ig_intelligentsia
			ig_counts_as_marginal = no
		}
	}

	cooldown = { days = normal_modifier_time }

	immediate = {
		currently_enacting_law = {
			save_scope_as = gun_law_scope
		}
		random_interest_group = {
			limit = {
				is_interest_group_type = ig_intelligentsia
				ig_counts_as_marginal = no
			}
			save_scope_as = intelligentsia_ig
			leader = {
				save_scope_as = intelligentsia_leader
			}
		}
	}

	# Option A: Embrace the public health framing — the data is on our side
	option = {
		name = gun_laws.1.a
		default_option = yes
		add_enactment_modifier = {
			name = gun_law_safety_argument
			# = 2
		}
		scope:intelligentsia_ig = {
			add_modifier = {
				name = gun_law_intelligentsia_engaged
				days = normal_modifier_time
			}
		}
		ai_chance = {
			base = 60
			modifier = {
				trigger = {
					exists = ruler
					ruler = {
						OR = {
							has_ideology = ideology_progressive_leader
							has_ideology = ideology_liberal_leader
						}
					}
				}
				add = 20
			}
		}
	}

	# Option B: Frame it as a law and order issue — keep the public health advocates out
	option = {
		name = gun_laws.1.b
		add_enactment_modifier = {
			name = gun_law_safety_argument
		}
		scope:intelligentsia_ig = {
			add_modifier = {
				name = gun_law_intelligentsia_sidelined
				days = normal_modifier_time
			}
		}
		ai_chance = { base = 40 }
	}
}
