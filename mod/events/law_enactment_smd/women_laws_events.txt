# ==========================================
# womens_rights_laws.1
# "The Letter of the Law"
# Fires when enacting legal protections.
# Lawyers and reformers argue over how
# far the legislation should reach.
# checkpoint_advance
# ==========================================

womens_rights_laws.1 = {
	type = country_event
	placement = ROOT

	title = womens_rights_laws.1.t
	desc = womens_rights_laws.1.d
	flavor = womens_rights_laws.1.f

	category = enactment

	event_image = {
		video = "unspecific_politicians_arguing"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_election.dds"

	duration = 3

	trigger = {
		is_enacting_law = law_type:law_legal_protections
		NOT = { has_modifier = womens_rights_legal_scope }
		any_interest_group = {
			is_interest_group_type = ig_intelligentsia
			ig_counts_as_marginal = no
		}
	}

	cooldown = { days = normal_modifier_time }

	immediate = {
		currently_enacting_law = {
			save_scope_as = womens_rights_law_scope
		}
		random_interest_group = {
			limit = {
				is_interest_group_type = ig_intelligentsia
				ig_counts_as_marginal = no
			}
			save_scope_as = intelligentsia_ig
			leader = {
				save_scope_as = intelligentsia_leader
			}
		}
	}

	# Option A: Broad protections — write them into every institution
	option = {
		name = womens_rights_laws.1.a
		default_option = yes
		add_enactment_modifier = {
			name = womens_rights_legal_scope
		}
		scope:intelligentsia_ig = {
			add_modifier = {
				name = womens_rights_intelligentsia_engaged
				days = normal_modifier_time
			}
		}
		ai_chance = {
			base = 60
			modifier = {
				trigger = {
					exists = ruler
					ruler = {
						OR = {
							has_ideology = ideology_progressive_leader
							has_ideology = ideology_liberal_leader
						}
					}
				}
				add = 20
			}
		}
	}

	# Option B: Narrow protections — limit scope to avoid a constitutional fight
	option = {
		name = womens_rights_laws.1.b
		add_enactment_modifier = {
			name = womens_rights_legal_scope
		}
		add_enactment_modifier = {
			name = womens_rights_narrow_scope
		}
		scope:intelligentsia_ig = {
			add_modifier = {
				name = womens_rights_intelligentsia_disappointed
				days = normal_modifier_time
			}
		}
		ai_chance = { base = 40 }
	}
}
