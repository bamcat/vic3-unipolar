# ── MANI PULITE.1 — A ROUTINE ARREST ─────
# Mario Chiesa, chairman of a Milan old
# people's home, is caught with a bribe.
# He tries to flush it down the toilet.
# The opening crack.

mani_pulite.1 = {
	type = country_event
	placement = ROOT

	title = mani_pulite.1.t
	desc = mani_pulite.1.d
	flavor = mani_pulite.1.f

	event_image = { video = "bodycam" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	duration = 3

	trigger = {
		c:ITA ?= this
		has_journal_entry = je_tangentopoli
		NOT = { has_variable = chiesa_arrested }
	}

	immediate = {
		set_variable = chiesa_arrested
	}

	option = {
		name = mani_pulite.1.a
		# The arrest is dismissed as a minor local matter
		add_modifier = {
			name = tangentopoli_chiesa_arrest
			months = 12
		}
		change_variable = { name = tangentopoli_exposure add = 5 }
	}
}

# ── MANI PULITE.2 — THE FIRST CONFESSIONS ─
# Facing prison, minor functionaries begin
# to talk. Each name leads to another.
# The system is starting to unravel.

mani_pulite.2 = {
	type = country_event
	placement = ROOT

	title = mani_pulite.2.t
	desc = mani_pulite.2.d
	flavor = mani_pulite.2.f

	event_image = { video = "middleeast_courtroom_upheaval" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	duration = 3

	trigger = {
		c:ITA ?= this
		has_journal_entry = je_tangentopoli
		has_variable = chiesa_arrested
		var:tangentopoli_exposure >= 20
		NOT = { has_variable = first_confessions }
	}

	immediate = {
		set_variable = first_confessions
		random_interest_group = {
			limit = {
				is_in_government = yes
				is_powerful = yes
			}
			save_scope_as = compromised_ig
		}
	}

	option = {
		name = mani_pulite.2.a
		# Accept the confessions — let the investigation widen
		change_variable = { name = tangentopoli_exposure add = 10 }
		scope:compromised_ig = {
			add_modifier = {
				name = tangentopoli_ig_under_scrutiny
				months = 24
			}
		}
		add_modifier = {
			name = tangentopoli_widening_investigation
			months = 24
		}
	}

	option = {
		name = mani_pulite.2.b
		# Try to contain it — use authority to suppress the spread
		trigger = {
			country_authority_add >= 150
		}
		change_variable = { name = tangentopoli_exposure add = 3 }
		add_modifier = {
			name = tangentopoli_suppression_attempt
			months = 12
		}
		country_add_legitimacy = -10
	}
}

# ── MANI PULITE.6 — AVVISI DI GARANZIA ───
# Notices of investigation arrive for
# politicians and businessmen across Italy.
# "Avviso di garanzia" becomes shorthand
# for disgrace.

mani_pulite.6 = {
	type = country_event
	placement = ROOT

	title = mani_pulite.6.t
	desc = mani_pulite.6.d
	flavor = mani_pulite.6.f

	event_image = { video = "europenorthamerica_springtime_of_nations" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	duration = 3

	trigger = {
		c:ITA ?= this
		has_journal_entry = je_mani_pulite
		NOT = { has_variable = avvisi_garanzia_fired }
	}

	immediate = {
		set_variable = avvisi_garanzia_fired
		random_interest_group = {
			limit = {
				is_in_government = yes
			}
			save_scope_as = indicted_ig
			leader = { save_scope_as = indicted_leader }
		}
	}

	option = {
		name = mani_pulite.6.a
		# Cooperate with investigators
		default_option = yes
		change_variable = { name = investigation_progress add = 8 }
		scope:indicted_ig = {
			add_modifier = {
				name = tangentopoli_ig_under_scrutiny
				months = 24
			}
		}
		country_add_legitimacy = -5
	}

	option = {
		name = mani_pulite.6.b
		# Resist — use executive pressure
		trigger = { country_authority_add >= 100 }
		change_variable = { name = political_resistance add = 10 }
		change_variable = { name = investigation_progress add = 2 }
		add_modifier = {
			name = tangentopoli_suppression_attempt
			months = 12
		}
		any_scope_state = {
			limit = {
				is_incorporated = yes
				turmoil >= 0.05
			}
			add_radicals_in_state = { value = small_radicals }
		}
	}
}

# He flees to Tunisia. The Socialists

				name = tangentopoli_ig_under_scrutiny
				months = 24
			}
		}

			limit = { is_incorporated = yes }

# ── MANI PULITE.8 — TANGENTOPOLI SPREADS ─
# The investigation reaches the South.
# Construction contracts, EU agricultural
# subsidies, public works — the system
# was nationwide.

mani_pulite.8 = {
	type = country_event
	placement = scope:tangentopoli_south_state

	title = mani_pulite.8.t
	desc = mani_pulite.8.d
	flavor = mani_pulite.8.f

	event_image = { video = "unspecific_vandalized_storefront" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	duration = 3

	trigger = {
		c:ITA ?= this
		has_journal_entry = je_mani_pulite
		var:investigation_progress >= 45
		NOT = { has_variable = tangentopoli_south }
		any_scope_state = {
			OR = {
				state_region = s:STATE_SICILY
				state_region = s:STATE_CALABRIA
			}
		}
	}

	immediate = {
		set_variable = tangentopoli_south
		random_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_SICILY
					state_region = s:STATE_CALABRIA
				}
			}
			save_scope_as = tangentopoli_south_state
		}
	}

	option = {
		name = mani_pulite.8.a
		# Extend the investigation
		change_variable = { name = investigation_progress add = 10 }
		scope:tangentopoli_south_state = {
			add_radicals_in_state = { value = medium_radicals strata = lower }
		}
		add_modifier = {
			name = tangentopoli_south_exposure
			months = 24
		}
	}

	option = {
		name = mani_pulite.8.b
		# Political resistance from southern clients
		change_variable = { name = political_resistance add = 8 }
		change_variable = { name = investigation_progress add = 3 }
		interest_group_ig_landowners_approval_add = -5
	}
}

# ── MANI PULITE.10 — THE FIRST REPUBLIC FALLS
# Investigation wins. The party system
# of the post-war era is destroyed.
# New political forces emerge from the ruins.

mani_pulite.10 = {
	type = country_event
	placement = ROOT

	title = mani_pulite.10.t
	desc = mani_pulite.10.d
	flavor = mani_pulite.10.f

	event_image = { video = "libyan_society" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	duration = 3

	trigger = {
		# triggered by je_mani_pulite on_complete
	}

	immediate = {
		# Save the most powerful IG for the transition narrative
		random_interest_group = {
			limit = {
				is_in_government = yes
				is_powerful = yes
			}
			save_scope_as = old_guard_ig
		}
	}

	option = {
		name = mani_pulite.10.a
		# Embrace the clean break — accelerate reforms
		default_option = yes
		remove_modifier = tangentopoli_systemic_corruption
		remove_modifier = tangentopoli_legitimacy_crisis
		add_modifier = {
			name = mani_pulite_clean_hands_victory
			months = 60
		}
		scope:old_guard_ig = {
			add_modifier = {
				name = tangentopoli_party_collapse
				months = 48
			}
		}
		# The political earthquake creates both loyalists and radicals
		every_scope_state = {
			limit = { is_incorporated = yes }
			add_loyalists_in_state = { value = medium_radicals strata = middle }
			add_radicals_in_state = { value = small_radicals strata = lower }
		}
		country_add_legitimacy = 15
	}

	option = {
		name = mani_pulite.10.b
		# Manage the transition — controlled reform
		remove_modifier = tangentopoli_systemic_corruption
		add_modifier = {
			name = mani_pulite_cautious_transition
			months = 48
		}
		scope:old_guard_ig = {
			add_modifier = {
				name = tangentopoli_ig_under_scrutiny
				months = 36
			}
		}
		every_scope_state = {
			limit = { is_incorporated = yes }
			add_radicals_in_state = { value = small_radicals }
		}
		country_add_legitimacy = 5
	}
}
