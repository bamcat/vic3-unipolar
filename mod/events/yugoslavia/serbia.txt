namespace = yugoslav_collapse

# Begins / crisis cabinet
yugoslav_collapse.1 = {
	type = country_event
	title = yugoslav_collapse.1.t
	desc = yugoslav_collapse.1.d
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	placement = ROOT
	event_image = { video = "europenorthamerica_before_the_battle" }
	duration = 2

	trigger = {
		c:YUG ?= this
		has_journal_entry = je_collapse_of_yugoslavia
		NOT = { has_variable = yugoslav_collapse_started }
	}

	immediate = {
		save_scope_as = yugoslav_collapse_root
		set_variable = yugoslav_collapse_started
	}

	option = {
		name = yugoslav_collapse.1.a
		default_option = yes
		custom_tooltip = begin_disintegration_tt
		yugo_increase_pressure_small = yes
		trigger_event = { id = yugoslav_collapse.2 days = 5 }
		trigger_event = { id = yugoslav_collapse.3 days = 10 }
		ai_chance = { base = 100 }
	}

	option = {
		name = yugoslav_collapse.1.b
		custom_tooltip = yugoslav_confederal_offer_tt
		yugo_reduce_pressure_large = yes
		add_modifier = { name = yugoslav_confederal_negotiations days = 730 }
		trigger_event = { id = yugoslav_collapse.10 days = 30 }
		ai_chance = { base = 0 }
	}

	option = {
		name = yugoslav_collapse.1.c
		custom_tooltip = yugoslav_federal_crackdown_tt
		yugo_increase_pressure_large = yes
		add_radicals = { value = 0.05 culture = culture:south_slavic }
		trigger_event = { id = yugoslav_collapse.2 days = 5 }
		trigger_event = { id = yugoslav_collapse.3 days = 10 }
		ai_chance = { base = 0 }
	}
}
