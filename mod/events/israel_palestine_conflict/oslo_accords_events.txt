namespace = oslo_accords

# Oslo I Accord / Declaration of Principles
# Step 1: Israel proposes the agreement.
oslo_accords.1 = {
	type = country_event
	placement = ROOT
	duration = 3

	title = oslo_accords.1.t
	desc = oslo_accords.1.d
	flavor = oslo_accords.1.f

	event_image = {
		video = "africa_diplomats_negotiating"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_diplomacy.dds"

	trigger = {
		game_date >= 1993.8.30
		c:ISR ?= this
		exists = c:PAL
		NOT = { has_variable = op1_p }
		NOT = { has_global_variable = oslo_i_signed }
		NOT = { has_global_variable = oslo_i_rejected }
	}

	immediate = { set_variable = op1_p }
	
	option = {
		name = oslo_accords.1.a
		default_option = yes

		custom_tooltip = { text = oslo_i_proposal_sent_tt }

		set_global_variable = oslo_i_proposed

		c:PAL ?= {
			trigger_event = {
				id = oslo_accords.10
				popup = yes
				days = 1
			}
		}
	}

	option = {
		name = oslo_accords.1.b

		custom_tooltip = { text = oslo_i_israel_delay_tt }

		set_global_variable = oslo_i_israel_declined

		add_modifier = {
			name = oslo_talks_delayed
			years = 1
		}
		c:PAL ?= {
			add_modifier = {
				name = oslo_talks_delayed
				years = 1
			}
		}

		custom_tooltip = { text = ip_temperature_increase_10_tt }
		ip_temperature_add_10 = yes
	}
}

# Oslo I Accord
# Step 2: Palestine accepts or rejects.
oslo_accords.10 = {
	type = country_event
	placement = ROOT
	duration = 3

	title = oslo_accords.10.t
	desc = oslo_accords.10.d
	flavor = oslo_accords.10.f

	event_image = {
		video = "africa_diplomats_negotiating"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_diplomacy.dds"

	trigger = {
		c:PAL ?= this
		exists = c:ISR
		has_global_variable = oslo_i_proposed
		NOT = { has_global_variable = oslo_i_signed }
		NOT = { has_global_variable = oslo_i_rejected }
	}

	option = {
		name = oslo_accords.10.a
		default_option = yes

		custom_tooltip = { text = oslo_i_palestine_accept_tt }
		custom_tooltip = { text = ip_temperature_decrease_25_tt }
		custom_tooltip = { text = ip_negotiations_increase_35_tt }
		custom_tooltip = { text = ip_international_pressure_decrease_10_tt }

		oslo_i_acceptance_effect = yes
		ai_chance = { base = 100 }
		c:ISR ?= {
			trigger_event = {
				id = oslo_accords.11
				popup = yes
				days = 1
			}
		}

		c:USA ?= {
			trigger_event = {
				id = oslo_accords.30
				popup = yes
				days = 1
			}
		}
		c:NRW ?= {
			trigger_event = {
				id = oslo_accords.30
				popup = yes
				days = 1
			}
		}
	}

	option = {
		name = oslo_accords.10.b

		custom_tooltip = { text = oslo_i_palestine_reject_tt }
		custom_tooltip = { text = ip_temperature_increase_20_tt }
		custom_tooltip = { text = ip_negotiations_decrease_20_tt }

		set_global_variable = oslo_i_rejected
		remove_global_variable = oslo_i_proposed

		add_modifier = {
			name = oslo_talks_delayed
			years = 2
		}
		c:ISR ?= {
			add_modifier = {
				name = oslo_talks_delayed
				years = 2
			}
			trigger_event = {
				id = oslo_accords.12
				popup = yes
				days = 1
			}
		}

		c:USA ?= {
			trigger_event = {
				id = oslo_accords.31
				popup = yes
				days = 1
			}
		}
		c:NRW ?= {
			trigger_event = {
				id = oslo_accords.31
				popup = yes
				days = 1
			}
		}

		ip_temperature_add_20 = yes
		if = {
			limit = { has_global_variable = ip_peacetime_negotiations }
			change_global_variable = { name = ip_peacetime_negotiations add = -20 }
		}
	}
}
