# Major Offensive / Mass Resistance Launched
intifada_events.1 = {
	type = country_event
	placement = ROOT
	duration = 3
	title = intifada_events.1.t
	desc = intifada_events.1.d
	event_image = { video = "europenorthamerica_before_the_battle" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	trigger = {}

	immediate = {
		if = {
			limit = { c:PAL ?= this }
			show_as_tooltip = {
				add_modifier = {
					name = intifada_popular_resistance
					multiplier = { value = global_var:pal_intifada_score min = 1 }
					years = 1
					is_decaying = yes
				}
			}
		}
		if = {
			limit = { c:ISR ?= this }
			show_as_tooltip = {
				add_modifier = {
					name = intifada_isr_morale
					multiplier = { value = global_var:isr_intifada_score min = 1 }
					years = 2
					is_decaying = yes
				}
			}
		}
	}

	option = {
		name = { trigger = { c:ISR ?= this } text = intifada_events.1.a }
		name = { trigger = { c:PAL ?= this } text = intifada_events.1.a.pal }
		default_option = yes
	}
}

# Stone Throwing / Street Clashes — Palestinian resistance event
intifada_events.5 = {
	type = country_event
	placement = ROOT
	duration = 3
	cooldown = { months = 4 }
	title = intifada_events.5.t
	desc = intifada_events.5.d
	event_image = { video = "unspecific_vandalized_storefront" }

	trigger = { c:PAL ?= this is_at_war = yes }

		name = intifada_events.5.a # Escalate the resistance
		default_option = yes

			}
		}
		ai_chance = { base = 8 }
	}

	option = {
		name = intifada_events.5.b # Show restraint
		ai_chance = { base = 2 }
	}
}

# International Condemnation — unique intifada event
# Media coverage shifts world opinion, damages Israel's reputation
intifada_events.8 = {
	type = country_event
	placement = ROOT
	duration = 3
	cooldown = { months = 12 }
	title = intifada_events.8.t
	desc = intifada_events.8.d
	event_image = { video = "africa_diplomats_negotiating" }
	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	trigger = {
		c:PAL ?= this
		is_at_war = yes
	}

	option = {
		name = intifada_events.8.a # Appeal to the world
		default_option = yes
		# International sympathy shifts toward Palestine
		c:ISR ?= {
			change_infamy = 3
			add_modifier = { name = intifada_international_pressure months = 12 }
		}
		# Arab League members improve relations with Palestine
		every_country = {
			limit = {
				is_target_in_global_variable_list = { name = io_arab_league_members target = this }
				NOT = { c:PAL ?= this }
			}
			change_relations = { country = c:PAL value = 10 }
			change_relations = { country = c:ISR value = -5 }
		}
		ai_chance = { base = 10 }
	}
}
