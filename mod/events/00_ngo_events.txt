	duration = 3

	trigger = {
		NOT = { has_variable = io_ngo_amnesty_banned }
		has_variable = io_ngo_amnesty_opinion

			has_law = law_type:law_outlawed_dissent
			has_law = law_type:law_secret_police
			has_law = law_type:law_active_oppression
			has_law = law_type:law_ethnostate
			has_law = law_type:law_slave_trade
			has_law = law_type:law_debt_slavery
			has_law = law_type:law_no_womens_rights
			has_law = law_type:law_militarized_police
			has_law = law_type:law_total_prohibition
			has_law = law_type:law_autocracy
		}
		NOT = { has_variable = ngo_amnesty_law_cooldown }
	}

	option = {
		name = ngo_events.1.a # Ignore the criticism
		default_option = yes
		custom_tooltip = {
			text = ngo_amnesty_opinion_minus_15_tt
			change_variable = { name = io_ngo_amnesty_opinion add = -15 }
			clamp_variable = { name = io_ngo_amnesty_opinion min = 0 max = 100 }
		}
		set_variable = { name = ngo_amnesty_law_cooldown years = 2 }
	}

	option = {
		name = ngo_events.1.b # Engage with Amnesty's concerns
		custom_tooltip = {
			text = ngo_amnesty_opinion_minus_5_tt
			change_variable = { name = io_ngo_amnesty_opinion add = -5 }

		}
		add_modifier = { name = ngo_amnesty_scrutiny months = 12 }
		set_variable = { name = ngo_amnesty_law_cooldown years = 2 }
	}
}
