great_polarization.30 = {
	type = country_event
	placement = ROOT
	title = great_polarization.30.t
	desc = great_polarization.30.d
	flavor = great_polarization.30.f
	event_image = { video = "southamerica_war_civilians" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_protest.dds"
	duration = 3

	trigger = {
		c:USA ?= this
		has_journal_entry = je_the_great_polarization
		game_date >= 2020.5.25
		NOT = { has_variable = george_floyd_event_happened }
		NOT = { has_global_variable = second_american_civil_war_started }
	}

	immediate = {
		set_variable = george_floyd_event_happened
	}

	option = {
		name = great_polarization.30.a
		default_option = yes
		add_modifier = {
			name = great_polarization_minorities_targeted
			days = 730
			is_decaying = yes
		}
		change_variable = { name = great_polarization_left_pressure add = 18 }
		great_polarization_raise_tension_large = yes
		ai_chance = { base = 100 }
	}
	option = {
		name = great_polarization.30.b
		add_modifier = {
			name = great_polarization_reform_concessions
			days = 730
			is_decaying = yes
		}
		change_variable = { name = great_polarization_left_pressure add = 6 }
		great_polarization_reduce_tension_medium = yes
		ai_chance = { base = 0 }
	}
}
