namespace = great_polarization

great_polarization.100 = {
	type = country_event
	placement = ROOT
	title = great_polarization.100.t
	desc = great_polarization.100.d
	flavor = great_polarization.100.f
	event_image = { video = "votp_barricade" }
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/event_war.dds"
	duration = 3

	trigger = {
		c:USA ?= this
		has_global_variable = second_american_civil_war_started
		NOT = { has_variable = second_american_civil_war_event_seen }
	}

	immediate = {
		set_variable = second_american_civil_war_event_seen
	}

	option = {
		name = great_polarization.100.a
		default_option = yes
		great_polarization_begin_second_american_civil_war = yes
		ai_chance = { base = 100 }
	}
}
